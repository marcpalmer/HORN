 62     var handleTemplateBinding = this.scope( function( node, path, components ) {
 63         var key;
 64         var componentData = this.getBindingData( node, path);
 65         if ( componentData !== false ) {
 66             key = this.pathIndicator({n: node});
 67             if ( this.isAdjustingPath( key) ) {
 68             componentData.path = (path + '-' + key); }
 69             if ( !componentData.isJSON  ) {
 70             components.push( componentData); }
 71             return false;
 72         }
 73         return true;

 76     /**
 77      * @private
 78      * @function
 79      */
 80     var getModelReference = this.scope( function( args ) {
 81         var rv;
 82         var tokens = this.pathToTokens( args);
 83         var length = tokens.length;
 84         if ( length > 0 ) {
 85             rv = {ref: state.model, key: tokens[ length - 1]};
 86             tokens.length = tokens.length - 1;
 87             this.each( tokens, function( i, n ) {
 88                 if ( this.isDefinedNotNull( rv.ref) ) {
 89                     if ( rv.ref.hasOwnProperty( n) ) { rv.ref = rv.ref[ n]; }
 90                 } else { return false; }
 91             }, this);
 92             if ( !this.isDefinedNotNull( rv.ref) ) { rv = undefined; }
 93         }
 94         return rv;
 95     }, this);

 97     /**
 98      * @private
 99      * @function
100      */
101     var setValue = this.scope( function( value, path, parentContext ) {
102         var token;
103         var numTokens;
104         var subContext;
105         if ( typeof path === 'string' ) {
106             path = this.pathToTokens( {path: path});
107             if ( !this.isDefinedNotNull( state.model) ) {
108                 state.model = (!isNaN( parseInt( path[ 0])) ? [] : {});
109             }
110             parentContext = state.model;
111         }
112         numTokens = path.length;
113         if ( numTokens > 0 ) {
114             token = path.shift();
115             if ( numTokens > 1 ) {
116                 if ( !parentContext.hasOwnProperty( token) ) {
117                     subContext = !isNaN( parseInt( path[ 0])) ? [] : {};
118                     parentContext[ token] = subContext;
119                 }
120                 subContext = parentContext[ token];
121                 return setValue( value, path, subContext);
122             } else {
123                 parentContext[ token] = value;
124                 return {context: parentContext, key: token, value: value};
125             }
126         }
127     }, this);

133     var convert = this.scope( function ( args ) {
134         var converter = state.opts.converter;
135         if ( this.startsWith( args.path, "-") ) {
136             args.path = args.path.substring( 1);
137         }
138         return this.isDefinedNotNull( converter) ?
139             converter.call( this, args) : undefined;

142     /**
143      * @private
144      * @function
145      */
146     var addBinding = this.scope( function( args ) {
147         var details;
148         var culledPath = args.path.substring( 1);
149         var rv;
150         if ( args.setModel !== false ) {
151             details = setValue(  args.value, args.path);
152         }
153         if ( (args.readOnly === false) && (!args.isJSON) ) {
154             if ( this.isDefinedNotNull( details) ) {
155                 args.context = details.context;
156                 args.key = details.key;
157             }
158 
159             if ( !this.isDefinedNotNull( state.bindings) ) {
160                 state.bindings = {}; }
161             rv = {node: args.node, value: args.value};
162             if ( this.isDefinedNotNull( args.context) ) {
163                 rv.context = args.context;
164                 rv.key = args.key;
165             }
166             state.bindings[ culledPath] = rv;
167         }
168     }, this);

170     /**
171      * @private
172      * @function
173      */
174     var addBindings = this.scope( function( args ) {
175         this.each(
176             args.components,
177             function( i, newArgs ) {
178                 var modelValue;
179                 var textValue;
180                 var ref = getModelReference( newArgs);
181                 if ( this.isDefinedNotNull( ref) )  {
182                     modelValue = ref.ref[ ref.key];
183                     textValue = convert( {
184                         value: modelValue,
185                         path:  newArgs.path,
186                         type:  'toText',
187                         node:  newArgs.node
188                     });
189                     if ( (textValue === undefined) &&
190                         this.isDefinedNotNull( modelValue)) {
191                         textValue = modelValue + "";
192                     }
193                     newArgs.text = textValue;
194                     this.setDOMNodeValue( {node: newArgs.node,
195                         value: newArgs.text});
196                     addBinding({
197                         setModel: false,
198                         node: newArgs.node,
199                         context: ref.ref,
200                         key: ref.key,
201                         value: modelValue,
202                         path: newArgs.path});
203                 }
204             }, this);
205     }, this);

207     /**
208      * @private
209      * @function
210      */
211     var addJSONBindings = this.scope(
212         function( args ) {
213             var defaults = {type:  'fromJSON', node:  args.node,
214                 readOnly: args.readOnly};
215             var addJSONHelper = this.scope( function( vargs ) {
216                 var oldValue = vargs.value;
217                 vargs.value = convert( vargs);
218                 if ( !this.isDefinedNotNull( vargs.value) ) {
219                     vargs.value = oldValue;
220                 }
221                 addBinding( vargs);
222             }, this);
223             var jsonData = $.evalJSON( args.text);
224             var rootPath = args.path;
225             if ( typeof jsonData === 'object' ) {
226                 this.traverse( jsonData,
227                     this.scope( function( k, v ) {
228                         addJSONHelper( $.extend( defaults,
229                             {value: v, path:  rootPath + k}));
230                     }, this), '');
231             } else {
232                 addJSONHelper( $.extend( defaults,
233                     {value: jsonData, path:  rootPath}));
234             }
235         }, this);

237     /**
238      * @private
239      * @function
240      */
241     var renderComponent = this.scope( function( args ) {
242         var rootNode = args.rootNode;
243         var component = args.component;
244         var modelValue = component.context[ component.key];
245         var textValue;
246         if ( modelValue !== component.value ) {
247             if ( !rootNode || (rootNode &&
248                 this.contains( $(component.node).parents(), rootNode)) ) {
249                 textValue = convert( {
250                     value: modelValue,
251                     path:  args.path,
252                     type:  'toText',
253                     node: component.node});
254                 if ( !this.isDefinedNotNull( textValue) ) {
255                     textValue = modelValue + "";
256                 }
257                 this.setDOMNodeValue( {node: component.node, value: textValue});
258                 component.value = modelValue;
259                 return component.node;
260             }
261         }
262     }, this);

264     /**
265      * @private
266      * @function
267      */
268     var extract = this.scope( function( args ) {
269         var _this = this;
270         var rootNodes = this.definesProperty( args, 'rootNodes') ?
271             args.rootNodes : (this.definesProperty( args, 'selector') ?
272                 $(args.selector) : this.rootNodes());
273         setDefaultModel();
274         this.each( rootNodes,
275             function( i, n ) { this.visitNodes( n, '',
276                 function( n, path ) {
277                     if ( _this.hasRootIndicator( { n: n}) ) {
278                         return false; }
279                     var componentData = _this.getBindingData( n, path);
280                     if ( componentData === false ) {
281                         return true; }
282 
283                     componentData.readOnly = args.readOnly;
284                     if ( componentData.isJSON === false ) {
285                         componentData.value = convert( {
286                             value: componentData.text,
287                             path:  componentData.path,
288                             type:  'fromText',
289                             node:  componentData.node
290                         });
291                         if ( !_this.isDefinedNotNull( componentData.value) ) {
292                             componentData.value = componentData.text;
293                         }
294                         addBinding( componentData);
295                     } else {
296                         addJSONBindings( componentData);
297                     }
298                     return false;
299                 }
300             ); }, this);
301         return state.model;
302     }, this);

355     /**
356      *  Create a new UI element by cloning an existing template that is
357      *  marked up with HORN indicators, and populate the DOM nodes with
358      *  data from the specified property path.
359      *
360      *  @param {Object} [args] all arguments for this function
361      *  @param {String} args.path the property path within the model, to use to populate
362      *      this DOM node and its descendants
363      *  @param [args.template] a jQuery object representing the DOM template to clone
364      *  @param [args.id] the new "id" attribute value for the cloned DOM node
365      *  @param {String} [args.selector]
366      *
367      *  @return the newly cloned and populated template
368      *
369      *  @public
370      */
371     this.cloneAndBind = function( args ) {
372         var template;
373         var pathStem;
374         var components = [];
375         if ( this.definesProperty( args, 'template') ) {
376             template = args.template;
377         } else {
378             template = $(args.selector).clone();
379             template.removeAttr( "id");
380         }
381 
382         if ( this.definesProperty( args, 'id') ) {
383             template.attr( "id", args.id);
384         }
385 
386         setDefaultModel();
387 
388         pathStem = this.definesProperty( args, 'path') ? args.path : '';
389 
390         this.visitNodes( template, pathStem,
391             this.scope( function( n, path ) {
392                 return this.handleTemplateBinding( n, path, components);
393             }, this));
394 
395         addBindings({components: components});
396 
397         return template;
398     };

414             var node = renderComponent( {rootNode: this.definesProperty(
415                 args, 'rootNode') ? args.rootNode : undefined,
416                     component: n, path: i});
417             if ( this.isDefinedNotNull( node) ) {
418                 alteredNodes.push( node); }
419         }, this);
420         return alteredNodes;
421     };

489      *  Shallow copies properties from source to destination objects.
490      *  <p>
491      *  Neither copies, <code>undefined</code> valued properties, nor
492      *  prototypical properties.
493      *  <p>
494      *  The source of property names to copy is given by args.dest's property
495      *  names unless the optional args.props is supplied in which case it is
496      *  used instead.
497      *
498      *  @param {Object} args all arguments for this function
499      *  @param {Object} args.src the property source
500      *  @param {Object} args.dest the property destination
501      *  @param {Object} [args.props] an alternative source of property names
502      *
503      *  @methodOf Horn.prototype

506         var val;
507         this.each( this.isDefinedNotNull( args.props) ?
508             args.props : args.dest, function( i, n ) {
509             if ( args.src.hasOwnProperty( i) ) {
510                 val = args.src[ i];
511                 if ( val !== undefined ) { args.dest[ i] = val; }
512             } }, this);

554     /**
555      *  xxxx
556      *  <p>
557      *
558      *  @param node
559      *  @param parentPath
560      *
561      *  @return
562      *
563      *  @methodOf Horn.prototype
564      */
565     getBindingData: function( node, parentPath ) {
566         var theContained;
567         var nodeName;
568         var path = this.pathIndicator({n: node});
569         var contents = $($(node).contents());
570         var cd = {
571             isJSON: this.jsonIndicator({n: node}),
572             path: this.isAdjustingPath( path),
573             node: node};
574         var contentsSize = contents.size();
575         var isEmptyNode = contentsSize === 0;
576         if ( (contentsSize === 1) || (isEmptyNode && !cd.isJSON))  {
577             if ( !isEmptyNode ) { theContained = contents[0]; }
578             if ( cd.isJSON || cd.path ) {
579                 cd.path = cd.path ? (parentPath + '-' + path) : parentPath;
580                 nodeName = node.nodeName.toLowerCase();
581                 cd.isFormElementNode =
582                     (nodeName === 'input') || (nodeName === 'textarea');
583                 cd.isABBRNode = !cd.isFormElementNode &&
584                     (nodeName.toLowerCase() === "abbr");
585                 cd.isTextNode = !cd.isABBRNode && (isEmptyNode ||
586                     (theContained.nodeType === Node.TEXT_NODE));
587 
588                 if ( cd.isFormElementNode || cd.isTextNode || cd.isABBRNode ) {
589                     cd.text = this.getDOMNodeValue( {node: node});
590                     return cd;
591                 }
592             }
593         }
594         return false;
595     },

747      *  @param value the value to test
748      *  @param stem the candidate prefix for the given value
749      *
750      *  @return <code>true</code> if the given <code>String</code> is prefixed,
751      *      by the given stem, <code>false</code> otherwise
752      *
753      *  @methodOf Horn.prototype
754      */
755     startsWith: function ( value, stem ) {
756         return  (stem.length > 0) &&
757             ((value = value.match( "^" + stem)) !== null) &&
758                 (value.toString() === stem);
759     },

761     /**
762      *  xxxx
763      *  <p>
764      *
765      *  @param value
766      *  @param callback
767      *  @param key
768      *  @param context
769      *  @param propName
770      *
771      *  @methodOf Horn.prototype
772      */
773     traverse: function( value, callback, key, context, propName ) {
774         if ( (value instanceof Object) || (value instanceof Array) ) {
775             this.each( value, function( k, v ) { this.traverse( v, callback,
776                 key ? (key + '-' + k) : ("-" + k), value, k);
777             }, this);
778         } else { callback( key, value, context, propName); }
779     },

782      *  xxxx

785      *  @param dataElement

787      *  @param fn
788      *
789      *  @methodOf Horn.prototype
790      */
791     visitNodes: function( dataElement, path, fn ) {
792         var _path = this.pathIndicator({n: dataElement});
793         if ( this.isAdjustingPath( _path) ) { path = (path + '-' + _path); }
794         this.each( window.$(dataElement).children(), function( i, n ) {
795             if ( fn( n, path) ) { this.visitNodes( n, path, fn); }}, this);
796     }
