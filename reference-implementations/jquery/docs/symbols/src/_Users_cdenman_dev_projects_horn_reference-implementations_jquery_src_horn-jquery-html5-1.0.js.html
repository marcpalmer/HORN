 15 /**
 16  *  A Horn delegate implementation that extracts Horn data from custom HTML5
 17  *  data attributes.
 18  *  <P>
 19  *  Please refer to our online documentation for fuller details
 20  *  <a href="http://horn.io/">http://horn.io/</a>.
 21  *  <P>
 22  *  Use this function to create new <code>HornHTML5Features</code> instances,
 23  *  thus: <code>var hornHTML5Features = new HornHTML5Features();</code>.
 24  *  <P>
 25  *  Set it to use on a horn instance,
 26  *  <code>horn.delegate( hornHTML5Features);</code>.
 27  *
 28  *  @constructor
 29  *
 30  *  @return {HornHTML5Features} a newly initialised
 31  *      <code>HornHTML5Features</code> instance
 32  */
 33 function HornHTML5Features() {
 34 
 35     /**
 36      *  Determine if a given node possesses a Horn root node indicator
 37      *  (the data attribute 'horn').
 38      *  <P>
 39      *  In this implementation we return <code>true</code> if 'node' declares
 40      *  a data-attribute named 'horn' with any value whatsoever.
 41      *
 42      *  @param {Element} node the node to examine as to declaring a root
 43      *      indicator
 44      *
 45      *  @return <code>true</code> if 'node' does possess a Horn root node
 46      *      indicator (of any value), <code>false</code> otherwise.
 47      *
 48      *  @public
 49      */
 50     this.hasRootIndicator = function( node ) {
 51         var hornDeclaration = SMUtils.getDataAttr(
 52             node, HornHTML5Features.dataNameHorn);
 53         var jsonDeclaration = SMUtils.getDataAttr(
 54             node, HornHTML5Features.dataNameJSON);
 55         return (SMUtils.isDefinedNotNull( hornDeclaration) &&
 56             SMUtils.hasPrefix( hornDeclaration, "/")) ||
 57             (SMUtils.isDefinedNotNull( jsonDeclaration) &&
 58                 SMUtils.hasPrefix( jsonDeclaration, "/"));
 59     };
 60 
 61     /**
 62      *  Determine if a node declares the JSON indicator.
 63      *  <P>
 64      *  Nodes that declare this indicator are implicitly value nodes and contain
 65      *  literal JSON encoded as the single text element body value child  of
 66      *  declaring elements.
 67      *
 68      *  @param {Element} node the element that may declare the Horn JSON
 69      *      indicator
 70      *
 71      *  @return <code>true</code> if 'node' declares the Horn JSON indicator
 72      *
 73      *  @public
 74      */
 75      this.hasJSONIndicator = function( node ) {
 76         return SMUtils.isDefinedNotNull(
 77             SMUtils.getDataAttr( node, HornHTML5Features.dataNameJSON));
 78     };
 79 
 80     /**
 81      *  Extracts and returns the Horn path indicator for a given node.
 82      *  <P>
 83      *  In this, the HTML5 implementation, nodes can declare paths using
 84      *  data-attributes named, either 'horn', OR 'json'.
 85      *  <P>
 86      *  Paths are given in full JavaScript-like syntax so for example,
 87      *  <code>a.b[a].x[2][3</code>
 88      *
 89      *  @param {Element} node the node from which to extract the path indicator
 90      *
 91      *  @return {String|Boolean} if 'node' does have a Horn path indicator, it
 92      *      is returned in <code>String</code> form, else
 93      *      <code>Boolean false</code> is be returned
 94      *
 95      *  @public
 96      */
 97     this.pathIndicator = function( node ) {
 98         var hornDeclaration = SMUtils.getDataAttr(
 99             node, HornHTML5Features.dataNameHorn);
100         var jsonDeclaration = SMUtils.getDataAttr(
101             node, HornHTML5Features.dataNameJSON);
102         if ( jsonDeclaration === 'true' ) { jsonDeclaration = undefined; }
103         var declaration = SMUtils.isDefinedNotNull( hornDeclaration) ?
104             hornDeclaration : jsonDeclaration;
105         return SMUtils.isDefinedNotNull( declaration) ?
106             Horn.toInternalPath(declaration) : declaration;
107     };
108 
109     /**
110      *  Return all the current HTML document's Horn root nodes.
111      *  <P>
112      *  This implementation thus returns all nodes with a data-attribute named
113      *  'horn' with a value that starts with a forward-slash character '/'.
114      *
115      *  @return a list of this document's Horn root nodes
116      *
117      *  @public
118      */
119     this.rootNodes = function( args ) {
120         return $('[data-horn*="/"], [data-horn-json*="/"]');
121     };
122 }
123 
124 /**
125  *  The HTML5 data-attribute name for nodes declaring Horn property paths.
126  *
127  *  @public
128  */
129 HornHTML5Features.dataNameHorn = 'horn';
130 
131 /**
132  *  The HTML5 data-attribute name for nodes declaring Horn JSON data property
133  *  paths.
134  *
135  *  @public
136  */
137 HornHTML5Features.dataNameJSON = 'horn-json';
138 
139 if ( SMUtils.isDefinedNotNull( horn) ) { horn.delegate( new HornHTML5Features()); }
