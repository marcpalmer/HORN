 12 /**
 13  *  Used to create new <code>HornHTML5Features</code> instances, thus:
 14  *      <code>var hornHTML5Features = new HornHTML5Features();</code>.
 15  *
 16  *  @constructor
 17  *
 18  *  @return {HornHTML5Features} a newly initialised
 19  *      <code>HornHTML5Features</code> instance
 20  */
 21 function HornHTML5Features() {
 22 
 23     /**
 24      *  description
 25      *
 26      *  @param {Element} node
 27      *
 28      *  @return
 29      *
 30      *  @public
 31      */
 32     this.hasRootIndicator = function( node ) {
 33         var hornDeclaration = SMUtils.getDataAttr(
 34             node, HornHTML5Features.dataNameHorn);
 35         var jsonDeclaration = SMUtils.getDataAttr(
 36             node, HornHTML5Features.dataNameJSON);
 37         return (SMUtils.isDefinedNotNull( hornDeclaration) &&
 38             SMUtils.hasPrefix( hornDeclaration, "/")) ||
 39             (SMUtils.isDefinedNotNull( jsonDeclaration) &&
 40                 SMUtils.hasPrefix( jsonDeclaration, "/"));
 41     };
 42 
 43     /**
 44      *  description
 45      *
 46      *  @param {Element} node
 47      *
 48      *  @return
 49      *
 50      *  @public
 51      */
 52     this.hasJSONIndicator = function( node ) {
 53         return SMUtils.isDefinedNotNull(
 54             SMUtils.getDataAttr( node, HornHTML5Features.dataNameJSON));
 55     };
 56 
 57     /**
 58      *  description
 59      *
 60      *  @param {Element} node
 61      *
 62      *  @return
 63      *
 64      *  @public
 65      */
 66     this.pathIndicator = function( node ) {
 67         var hornDeclaration = SMUtils.getDataAttr(
 68             node, HornHTML5Features.dataNameHorn);
 69         var jsonDeclaration = SMUtils.getDataAttr(
 70             node, HornHTML5Features.dataNameJSON);
 71         if ( jsonDeclaration === 'true' ) { jsonDeclaration = undefined; }
 72         var declaration = SMUtils.isDefinedNotNull( hornDeclaration) ?
 73             hornDeclaration : jsonDeclaration;
 74         return SMUtils.isDefinedNotNull( declaration) ?
 75             Horn.toInternalPath(declaration) : declaration;
 76     };
 77 
 78     /**
 79      *  description
 80      *
 81      *  @param {Element} node
 82      *
 83      *  @return
 84      *
 85      *  @public
 86      */
 87     this.rootNodes = function( args ) {
 88         return $('[data-horn*="/"], [data-horn-json*="/"]');
 89     };
 90 }
 91 
 92 horn.delegate( new HornHTML5Features());
 93 
 94 /**
 95  *  Description
 96  *
 97  *  @public
 98  */
 99 HornHTML5Features.dataNameHorn = 'horn';
100 
101 /**
102  *  Description
103  *
104  *  @public
105  */
106 HornHTML5Features.dataNameJSON = 'horn-json';
