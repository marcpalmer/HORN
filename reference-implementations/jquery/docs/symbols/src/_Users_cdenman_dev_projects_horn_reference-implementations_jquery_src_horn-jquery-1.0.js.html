 28     /**
 29      *  Sets a model value given a path and value (if desired) and optionally
 30      *  adds a binding for the given: context, key, node, value and path.
 31      *
 32      *  @param {Object} args
 33      *  @param {Boolean} [args.setModel] default <code>true</code>, if
 34      *      <code>false</code> the model value isn't set, otherwise it is
 35      *  @param {Boolean} [args.readOnly] default <code>false</code>, if 
 36      *      <code>true</code> this binding is readOnly, otherwise it is two-way
 37      *  @param {Boolean} [args.isJSON] default <code>false</code>
 38      *  @param {String} [args.path]
 39      *  @param {Object} [args.context]
 40      *  @param {String} [args.key]
 41      *  @param {Element} [args.node]
 42      *  @param args.value
 43      *
 44      *  @private
 45      *  @function
 46      */
 47     var setModelAddBinding = SMUtils.bind( function( args ) {
 48         var rv;
 49         var details;
 50         if ( args.setModel !== false ) {
 51             details = setValue(  args.value, args.path);
 52         }
 53         if ( (args.readOnly !== true) && (args.isJSON !== true) ) {
 54             if ( SMUtils.isDefinedNotNull( details) ) {
 55                 args.context = details.context;
 56                 args.key = details.key;
 57             }
 58             if ( !SMUtils.isDefinedNotNull( state.bindings) ) {
 59                 state.bindings = {}; }
 60             rv = {node: args.node, value: args.value};
 61             if ( SMUtils.isDefinedNotNull( args.context) ) {
 62                 rv.context = args.context;
 63                 rv.key = args.key;
 64             }
 65             state.bindings[ args.path] = rv;
 66         }
 67     }, this);

 69     /**
 70      * @private
 71      * @function
 72      */
 73     var setModelAddBindings = SMUtils.bind( function( args ) {
 74         SMUtils.each(
 75             args.bindings,
 76             function( i, newArgs ) {
 77                 var modelValue;
 78                 var textValue;
 79                 var ref = getModelReference( newArgs);
 80                 if ( SMUtils.isDefinedNotNull( ref) )  {
 81                     modelValue = ref.ref[ ref.key];
 82                     textValue = convert( {
 83                         value: modelValue,
 84                         type:  'toText',
 85                         path:  newArgs.path,
 86                         node:  newArgs.node
 87                     });
 88                     if ( (textValue === undefined) &&
 89                         SMUtils.isDefinedNotNull( modelValue)) {
 90                         textValue = modelValue + "";
 91                     }
 92                     newArgs.text = textValue;
 93                     Horn.hornNodeValue( {node: newArgs.node,
 94                         value: newArgs.text});
 95                     setModelAddBinding({
 96                         setModel: false,
 97                         value: modelValue,
 98                         node: newArgs.node,
 99                         key: ref.key,
100                         context: ref.ref,
101                         path: newArgs.path});
102                 }
103             }, this);
104     }, this);

106     /**
107      * @private
108      * @function

110     var addJSONBindings = SMUtils.bind(
111         function( args ) {
112             var defaults = {type:  'fromJSON', node:  args.node,
113                 readOnly: args.readOnly};
114             var addJSONHelper = SMUtils.bind( function( vargs ) {
115                 var oldValue = vargs.value;
116                 vargs.value = convert( vargs);
117                 if ( !SMUtils.isDefinedNotNull( vargs.value) ) {
118                     vargs.value = oldValue;
119                 }
120                 setModelAddBinding( vargs);
121             }, this);
122             var jsonData = $.evalJSON( args.text);
123             if ( typeof jsonData === 'object' ) {
124                 Horn.traverse( jsonData,
125                     SMUtils.bind( function( k, v ) {
126                         addJSONHelper( $.extend( defaults, { value: v,
127                             path:  args.path + k})); }, this));
128             } else {
129                 addJSONHelper( $.extend( defaults,
130                     {value: jsonData, path:  args.path}));
131             }
132         }, this);

145     /**
146      * @private
147      * @function
148      */
149     var extract = SMUtils.bind( function( args ) {
150         var _this = this;
151         var pathStem = SMUtils.definesProperty( args, 'pathStem') ?
152             Horn.toInternalPath( args.pathStem) : undefined;
153         var rootNodes = SMUtils.definesProperty( args, 'nodes') ?
154             $(args.nodes) : delegate.rootNodes();
155         setDefaultModel();
156         SMUtils.each( rootNodes,
157             function( i, n ) {
158                 var inGraph = false;
159                 this.walkDOM( n,
160                     function( n, path ) {
161                         if ( delegate.hasRootIndicator(n) ) {
162                             if ( inGraph ) { return false; }
163                                 else { inGraph = true; }
164                         }
165                         var bindingData = _this.hasHornBinding( n);
166                         if ( bindingData === false ) {
167                             return true; }
168                         bindingData.readOnly = args.readOnly;
169                         bindingData.path = Horn.combinePaths( pathStem, path);
170                         if ( bindingData.isJSON === false ) {
171                             bindingData.value = convert( {
172                                 value: bindingData.text,
173                                 path:  bindingData.path,
174                                 type:  'fromText',
175                                 node:  bindingData.node
176                             });
177                             if ( !SMUtils.isDefinedNotNull( bindingData.value) ) {
178                                 bindingData.value = bindingData.text;
179                             }
180                             setModelAddBinding( bindingData);
181                         } else {
182                             addJSONBindings( bindingData);
183                         }
184                         return false;
185                     }
186                 );
187             }, this);
188         return state.model;
189     }, this);

191     /**
192      * @private
193      * @function
194      */
195     var getModelReference = SMUtils.bind( function( args ) {
196         var rv;
197         var tokens = Horn.pathToTokens( args.path);
198         var length = tokens.length;
199         if ( length > 0 ) {
200             rv = {ref: state.model, key: tokens[ length - 1]};
201             tokens.length = tokens.length - 1;
202             SMUtils.each( tokens, function( i, n ) {
203                 if ( SMUtils.isDefinedNotNull( rv.ref) ) {
204                     if ( rv.ref.hasOwnProperty( n) ) { rv.ref = rv.ref[ n]; }
205                 } else { return false; }
206             }, this);
207             if ( !SMUtils.isDefinedNotNull( rv.ref) ) { rv = undefined; }
208         }
209         return rv;
210     }, this);

218         if ( (bindingData !== false) && !bindingData.isJSON ) {
219             bindingData.path = path;
220             bindings.push( bindingData);
221             return false;
222         }
223         return true;

230     var render = SMUtils.bind( function( args ) {
231         var rootNode = args.rootNode;
232         var binding = args.binding;
233         var modelValue = binding.context[ binding.key];
234         var textValue;
235         var cArgs;
236         if ( args.forceRender === true ||
237             !SMUtils.compare( modelValue, binding.value) ) {
238             if ( !SMUtils.isDefinedNotNull( rootNode) ||
239                 SMUtils.compare( rootNode, binding.node) ||
240                 SMUtils.contains( $(binding.node).parents(), rootNode) ) {
241                 cArgs = {
242                     value: modelValue,
243                     path:  args.path,
244                     type:  'toText',
245                     node: binding.node};
246                 textValue = convert( cArgs);
247                 if ( !SMUtils.isDefinedNotNull( textValue) ) {
248                     textValue = modelValue + "";
249                 }
250                 Horn.hornNodeValue( {node: binding.node, value: textValue});
251                 binding.value = modelValue;
252                 return binding.node;
253             }
254         }
255     }, this);

281     /**
282      * @private
283      * @function
284      */
285     var setValue = SMUtils.bind( function( value, path, parentContext ) {
286         var token;
287         var numTokens;
288         var subContext;
289         if ( typeof path === 'string' ) {
290             path = Horn.pathToTokens( path);
291             if ( !SMUtils.isDefinedNotNull( state.model) ) {
292                 state.model = (!isNaN( parseInt( path[ 0])) ? [] : {});
293             }
294             parentContext = state.model;
295         }
296         numTokens = path.length;
297         if ( numTokens > 0 ) {
298             token = path.shift();
299             if ( numTokens > 1 ) {
300                 if ( !parentContext.hasOwnProperty( token) ) {
301                     subContext = !isNaN( parseInt( path[ 0])) ? [] : {};
302                     parentContext[ token] = subContext;
303                 }
304                 subContext = parentContext[ token];
305                 return setValue( value, path, subContext);
306             } else {
307                 parentContext[ token] = value;
308                 return {context: parentContext, key: token, value: value};
309             }
310         }
311     }, this);

313    /**
314      *  Inserts or deletes a bound model-array element.
315      *  <P>
316      *  Adjusts internal bindings to correct for the given operation.
317      *
318      *  @params args
319      *  @params {Boolean} [args.isInsert] (default <code>true</code>) are we
320      *      inserting into, or deleting from the given array
321      *  @params {String} args.path the path (into the model) of the array
322      *      element being added or removed
323      *
324      *  @public
325      */
326     this.adjustModelArray = function( args ) {
327         var prefix;
328         var prefixLength;
329         var postfix;
330         var alteredIndex;
331         var newBindings = {};
332         var path = Horn.toInternalPath( args.path);
333         var isInsert = args.isInsert === true;
334         var modelRef = getModelReference( {path: path});
335         var keyLength = (modelRef.key + "").length;
336         if ( SMUtils.isDefinedNotNull(modelRef) && $.isArray(modelRef.ref) ) {
337             prefix = SMUtils.replacePostfix( path, modelRef.key, "");
338             prefixLength = prefix.length;
339             postfix = path.substring( prefixLength + keyLength + 1);
340             alteredIndex = parseInt( modelRef.key);
341             SMUtils[isInsert ? 'arrayInsert' : 'arrayRemove']( modelRef.ref,
342                 alteredIndex);
343             SMUtils.each( state.bindings, function( bindingPath, n ) {
344                 var index;
345                 var indexStr;
346                 var hitUs;
347                 if ( bindingPath.substring( 0, prefixLength) === prefix ) {
348                     path = bindingPath.substring(prefixLength);
349                     index = path.indexOf( "-");
350                     indexStr = path.substring( 0, index !== -1 ?
351                         index : path.length);
352                     index = parseInt( indexStr);
353                     hitUs = index === alteredIndex;
354                     if ( hitUs && !isInsert ) { return; }
355                     if (index > alteredIndex) {
356                         isInsert ? index++ : index--;
357                     } else if (hitUs) {
358                         index++;
359                     }
360                     path = prefix + index + path.substring( indexStr.length);
361                     if ( n.key === indexStr ) { n.key = index + ""; }
362                 }
363                 newBindings[ path] = n;
364             }, this);
365         }
366         state.bindings = newBindings;
367     };

381     /**
382      *  Walk DOM tree(s) and extract model data, allowing for subsequent model
383      *  to DOM updates.
384      *  <p>
385      *  After execution, each Horn value element encountered will have a
386      *  corresponding representation in the model. Altering such model
387      *  values and then calling <code>updateDOM(...)</code> will refresh their
388      *  display values.
389      *  <p>
390      *  The DOM tree(s) to walk may be specified in exactly one of two ways:
391      *  <ol>
392      *      <li>
393      *          Horn will automatically find all DOM trees that have a root
394      *          indicator.
395      *      </li>
396      *      <li>
397      *          Passing in 'args.nodes', a collection of DOM elements or a
398      *          jQuery node selector.
399      *      </li>
400      *  </ol>
401      *  <p>
402      *  Before a value is stored in the model it is converted to its
403      *  <code>String</code> representation. Alternatively, applications can
404      *  register converter functions to override this default behaviour.
405      *  <p>
406      *  If model to DOM updates are not required, the alternative yet otherwise
407      *  identical function, <code>{@link Horn#load}</code> should be used
408      *  (or the 'readOnly' option).
409      *
410      *  @param {Object} args
411      *  @param {Element|String} [args.nodes] DOM nodes or a jQuery selector
412      *  @param {String} [args.pathStem] a property path prefix that will be
413      *      appended the paths of value nodes' property paths before storage in
414      *      the model
415      *
416      *  @return the updated model
417      *
418      *  @see Horn#option for the detailing of converter functions
419      *  @see Horn#load
420      *
421      *  @public
422      */
423     this.bind = function( args ) {
424         return extract(
425             setDefaultArgs( {args: args, defaults: {readOnly: false}}));
426     };

428     /**
429      *  Returns all <strong>bound</strong> model-entry paths that contain blank
430      *  <code>""</code> <code>String</code> values.
431      *  <p>
432      *  The model paths can be optionally filtered by requiring them to have a
433      *  given prefix.
434      *  <p>
435      *  Also optionally, the DOM nodes corresponding to the given model entry
436      *  can be content-checked to determine if it (instead) has a blank
437      *  <code>String</code> value.
438      *
439      *  @param {Object} args
440      *  @param {String} [args.path] the prefix that returned property paths must
441      *      have, if not supplied all property paths will be returned - if they
442      *      fulfill the content requirements
443      *  @param {Boolean} [args.inspectDOM] iff. <code>true<code> then instead
444      *      of using the model entries' value to compare with <code>""</code>,
445      *      the model entries' corresponding DOM node's value will be used
446      *      instead
447      *
448      *  @return {Array} an <code>Array</code> containing all model entries that
449      *      have the given (or any) path prefix and have either their model, or
450      *      DOM node values equal to the blank <code>String</code>
451      *      <code>""</code>
452      *
453      *  @public
454      */
455     this.blankModelEntries = function( args ) {
456         var blankPaths = [];
457         var pathDefined = SMUtils.definesProperty( args, 'path');
458         var inspectDOM = SMUtils.definesProperty( args, 'inspectDOM') &&
459             (args.inspectDOM === true);
460         var path = pathDefined ? Horn.toInternalPath( args.path) : undefined;
461         SMUtils.each( state.bindings, function( i, n ) {
462             if ( (!pathDefined || SMUtils.hasPrefix( i, path)) ) {
463                 if (((inspectDOM && SMUtils.isDefinedNotNull( n.node)) ?
464                     Horn.hornNodeValue( {node: n.node}) : n.value) === "") {
465                     blankPaths.push(Horn.toExternalPath(i));
466                 }
467             }
468         }, this);
469         return blankPaths;
470     };

472     /**
473      *  Walks a tree of cloned (or selected) DOM nodes, binding them to model
474      *  elements and populating their display values.
475      *  <p>
476      *  During the walk, the property path for each DOM value node visited is
477      *  inspected and if there is a corresponding model element with the same
478      *  path it will be bound to the DOM node and the DOM node will receive its
479      *  value.
480      *  <p>
481      *  For each DOM value node encountered, the optional 'args.pathStem' will
482      *  be added to its property path before binding to a model value. Allowing
483      *  nodes derived from a single template to be bound to multiple model
484      *  locations.
485      *  <p>
486      *  Model to DOM data conversions are performed using this function with the
487      *  type parameter value 'toText'.
488      *
489      *  @param {Object} args
490      *  @param {Element|String} [args.node] a jQuery node or selector - nodes
491      *      selected by this argument will not be cloned.
492      *  @param {Element|String} [args.template] jQuery node or selector - these
493      *      nodes will be cloned and the root node will have its 'id' attribute
494      *      removed and set to 'args.newID' if supplied
495      *  @param {String} [args.id] the new 'id' attribute value applied to nodes
496      *      cloned, if using 'args.template'
497      *  @param {String} [args.pathStem] a property path prefix appended to all
498      *      walked DOM nodes before binding to model values
499      *
500      *  @return the newly cloned and populated template
501      *
502      *  @public
503      */
504     this.bindTo = function( args ) {
505         var node;
506         var pathStem;
507         var bindings = [];
508         if ( SMUtils.definesProperty( args, 'node') ) {
509             node = $(args.node);
510         } else {
511             node = $(args.template).clone();
512             node.removeAttr( "id");
513             if ( SMUtils.definesProperty( args, 'id') ) {
514                node.attr( "id", args.id);
515             }
516         }
517         setDefaultModel();
518         pathStem = SMUtils.definesProperty( args, 'pathStem') ?
519             Horn.toInternalPath( args.pathStem) : '';
520         this.walkDOM( node,
521             SMUtils.bind( function( n, path ) {
522                 return handleTemplateBinding( n, path, bindings);
523             }, this), pathStem);
524         setModelAddBindings({bindings: bindings});
525         return node;
526     };

538     /**
539      *  Determines if a given node in the context of a Horn DOM tree is a value
540      *  node or not.
541      *  <p>
542      *  If the node is a value node, the necessary binding information is
543      *  extracted and returned.
544      *
545      *  @param node the DOM node to examine
546      *
547      *  @return <code>false</code> if the given node is not a value node else,
548      *      this function returns an object containing the binding information
549      *      extracted
550      *
551      *  @public
552      *
553      *  @todo move
554      */
555     this.hasHornBinding = function( node ) {
556         var theContained;
557         var nodeName;
558         var contents = $($(node).contents());
559         var isPathDefined = Horn.isPathDefined( delegate.pathIndicator(node));
560         var cd = {isJSON: delegate.hasJSONIndicator(node), node: node};
561         var contentsSize = contents.size();
562         var isEmptyNode = contentsSize === 0;
563         if ( (contentsSize === 1) || (isEmptyNode && !cd.isJSON))  {
564             if ( !isEmptyNode ) { theContained = contents[0]; }
565             if ( cd.isJSON || isPathDefined ) {
566                 nodeName = node.nodeName.toLowerCase();
567                 cd.isFormElementNode =
568                     (nodeName === 'input') || (nodeName === 'textarea');
569                 cd.isABBRNode = !cd.isFormElementNode &&
570                     (nodeName.toLowerCase() === "abbr");
571                 cd.isTextNode = !cd.isABBRNode && (isEmptyNode ||
572                     (theContained.nodeType === Node.TEXT_NODE));
573                 if ( cd.isFormElementNode || cd.isTextNode || cd.isABBRNode ) {
574                     cd.text = Horn.hornNodeValue( {node: node});
575                     return cd;
576                 }
577             }
578         }
579         return false;
580     };

630     /**
631      *  Get an option's value by name, or set an option's value by name.
632      *  <p>
633      *  If no value is provided, the value of the named option is returned,
634      *  otherwise the return value is undefined.
635      *  <p>
636      *  The following options are currently supported:<br>
637      *  <ul>
638      *      <li>
639      *          <p>
640      *          <strong>defaultModel</strong> - For setting an explicit default
641      *          model (<code>Object</code> or <code>Array</code>)
642      *          </p>
643      *      </li>
644      *      <li>
645      *          <p>
646      *          <strong>readOnly</strong> - If set to true, the automatic
647      *          extraction at startup will call load() instead of bind() so
648      *          there is no two-way binding to the DOM.
649      *          </p>
650      *      </li>
651      *      <li>
652      *          <p>
653      *          <strong>converter</strong> - A function with the following
654      *          signature, <code>(args)</code> that will be called upon to
655      *          convert values under the following scenarios:
656      *          </p>
657      *          <p>
658      *              <ul>
659      *                  <li>
660      *                      <p>
661      *                      <strong>DOM to model</strong> :- when values are
662      *                      extracted from the DOM for storage in the model
663      *                      (functions <code>{@link Horn#bind}</code>,
664      *                      <code>{@link Horn#load}</code>) the
665      *                      <code>convert</code> function
666      *                      will be called with 'type' parameter value
667      *                      '<strong>fromText</strong>'.
668      *                      </p>
669      *                  </li>
670      *                  <li>
671      *                      <p>
672      *                      <strong>Model to DOM</strong> :- when values
673      *                      read from the model are destined for the DOM,
674      *                      (functions <code>{@link Horn#updateDOM}</code>,
675      *                      <code>{@link Horn#bindTo}</code>) the
676      *                      <code>convert</code> function will be called with
677      *                      'type' parameter value '<strong>toText</strong>'.
678      *                      </p>
679      *                  </li>
680      *                  <li>
681      *                      <p>
682      *                      <strong>JSON to model</strong> :- when JSON values
683      *                      are extracted from the DOM for model storage
684      *                      (functions <code>{@link Horn#bind}</code>,
685      *                      <code>{@link Horn#load}</code>), the
686      *                      <code>convert</code> function will be called with
687      *                      'type' parameter value '<strong>fromJSON</strong>'.
688      *                      </p>
689      *                  </li>
690      *              </ul>
691      *          </p>
692      *          <p>
693      *              The complete set of arguments passed to the
694      *              <code>convert</code> function are namely:
695      *              <ul>
696      *                  <li>
697      *                      <p>
698      *                      <strong>args.type</strong> :- the conversion 'type'
699      *                      as described above.
700      *                      </p>
701      *                  </li>
702      *
703      *                  <li>
704      *                      <p>
705      *                      <strong>args.value</strong> :- the value for
706      *                      conversion. The JavaScript type is dependent on the
707      *                      conversion being performed (<code>String</code> for
708      *                      '<strong>fromText</strong>' conversions and the
709      *                      corresponding model value's type in all others).
710      *                      </p>
711      *                  </li>
712      *
713      *                  <li>
714      *                      <p>
715      *                      <strong>args.path</strong> :- the property path for
716      *                      the DOM or model value.
717      *                      </p>
718      *                  </li>
719      *
720      *                  <li>
721      *                      <p>
722      *                      <strong>args.node</strong> :- the DOM node that
723      *                      generated the value.
724      *                      </p>
725      *                  </li>
726      *              </ul>
727      *          </p>
728      *          <p>
729      *              Converter functions can return <code>undefined</code> in
730      *              which case the value (all such values) will be treated as
731      *              <code>String</code> types - the same behaviour as having no
732      *              converter function registered.
733      *          </p>
734      *          <p>
735      *              <strong>Only a single converter function can be registered,
736      *              with any invocation overwriting any previously registered
737      *              function.</strong>
738      *          </p>
739      *      </li>
740      *  </ul>
741      *
742      *  @param {Object} optionName the name of the option to set
743      *  @param {Object} [value] the value to set
744      *
745      *  @return the value of the named option if no value supplied else
746      *      <code>undefined</code>
747      *
748      *  @public
749      */
750     this.option = function( optionName, value ) {
751         if ( value !== undefined ) {
752             state.opts[ optionName] = value;
753         } else { return state.opts[ optionName]; }
754     };

768     /**
769      *  Removes either, all bindings or, all bindings with a given property path
770      *  , or all bindings with property path's that match a given regular
771      *  expression.
772      *  <p>
773      *  If no arguments are defined, <strong>all bindings are removed</strong>.
774      *  <p>
775      *  If both 'args.path' and 'args.pattern' are supplied, the result is the
776      *  same as calling this function twice, once supplying 'args.path' and
777      *  thence 'args.pattern'.
778      *
779      *  @param {Object} args
780      *  @param {String} [args.path] all bindings with property path's that
781      *      equal this argument, will be removed
782      *  @param {String} [args.stem] all bindings with property path's that
783      *      start with this stem, will be removed
784      *  @param {String} [args.pattern] all bindings with property path's that
785      *      match this regular expression, will be removed
786      *
787      *  @public
788      */
789     this.unbind = function( args ) {
790         var definesPath = SMUtils.definesProperty( args, 'path');
791         var definesPattern = SMUtils.definesProperty( args, 'pattern');
792         var definesStem = SMUtils.definesProperty( args, 'stem');
793         var internalStem;
794         if ( definesStem ) { internalStem = Horn.toInternalPath( args.stem); }
795         var internalPath;
796         if ( definesPath ) { internalPath = Horn.toInternalPath( args.path); }
797         var unbindAll = !definesPattern && !definesPath && !definesStem;
798         SMUtils.each( state.bindings,
799             function( i, n ) {
800                 var externalPath;
801                 var match;
802                 var deleteBinding = unbindAll ||
803                     (definesPath && (i === internalPath)) ||
804                     (definesStem && (SMUtils.startsWith( i, internalStem)));
805                 if ( !deleteBinding && definesPattern ) {
806                     externalPath = Horn.toExternalPath( i);
807                     match = externalPath.match( args.pattern);
808                     deleteBinding = match &&
809                         (match.toString() === externalPath);
810                 }
811                 if ( deleteBinding ) { delete state.bindings[ i]; }
812             }, this);
813     };

815     /**
816      *  Update all bound DOM nodes with their current model values, if altered.
817      *  <p>
818      *  This function will not update DOM nodes if their model value has not
819      *  changed.
820      *
821      *  @param {Element} rootNode optional DOM node such that if supplied, only
822      *      nodes under this nodes will be updated
823      *  @param {Boolean} forceRender if set, nodes will be rendered regardless
824      *      of the state of th
825      *
826      *  @return {Array} an array of nodes that had their DOM values changed
827      *
828      *  @public
829      */
830     this.updateDOM = function( rootNode, forceRender ) {
831         var alteredNodes = [];
832         SMUtils.each( state.bindings, function( i, n ) {
833             var node = render( {rootNode: rootNode, binding: n, path: i,
834                 forceRender: forceRender});
835             if ( SMUtils.isDefinedNotNull( node) ) { alteredNodes.push( node); }
836         }, this);
837         return alteredNodes;
838     };

840     /**
841      *  Walks the DOM and executes a callback for each node visited, building
842      *  up a property paths as it goes.
843      *  <p>
844      *  This function takes a property path stem it prepends to all Horn paths
845      *  constructed.
846      *  <p>
847      *  The callback function can stop at a given node by returning a non
848      *  <code>true</code> value.
849      *
850      *  @param {Element} node the node to start the walk from, this node is
851      *      implicitly visited (the callback will not be executed in respect
852      *      of it)
853      *  @param {Function} callback the callback function with the following
854      *      signature (node, path)  - where node is the current node being
855      *      visited and path is its full property path (relative to the first
856      *      DOM node visited and the path argument to <code>walkDOM(...)</code>
857      *      proper
858      *  @param {String} [path] the Horn property path stem that will be
859      *      prepended to each Horn path constructed
860      *
861      *  @public
862      *
863      *  @todo use SMUtils.walkDOM
864      */
865     this.walkDOM = function( node, callback, path ) {
866         if ( !SMUtils.isDefinedNotNull( path) ) { path = ''; }
867         path = Horn.combinePaths( path, delegate.pathIndicator(node));
868         if ( callback( node, path) === true ) {
869             SMUtils.each( $(node).children(), function( i, n ) {
870                 this.walkDOM( n, callback, path); }, this);
871         }
872     };
873 
874     this.reset();
875 }

877 /**
878  *  Join parent and child (internal) property paths.
879  *  <p>
880  *  If exactly one path is <code>null</code> or <code>undefined</code>, the
881  *  other is returned. If both paths are not defined, the empty
882  *  <code>String</code> is returned.
883  *
884  *  @param {String} [parent] the parent horn property path
885  *  @param {String} [child] the child horn property path
886  *
887  *  @return {String} the resultant, combined property path
888  *
889  *  @public
890  */
891 Horn.combinePaths = function( parent, child ) {
892     var parentDefined = Horn.isPathDefined( parent);
893     var childDefined = Horn.isPathDefined( child);
894     if ( parentDefined && childDefined ) {
895         return parent + "-" + child;
896     } else if ( parentDefined ) {
897         return parent;
898     } else if ( childDefined ) {
899         return child;
900     } else { return ""; }
901 };

903 /**
904  *  Sets or retrieves a DOM node's Horn text.
905  *  <p>
906  *  The value retrieved is HTML un-escaped.
907  *
908  *  @param {Object} args
909  *  @param {Element} args.node the node from which to retrieve text
910  *  @param {Object} args.value the value to set
911  *
912  *  @return {String} the given node's displayed text
913  *
914  *  @public
915  */
916 Horn.hornNodeValue = function( args ) {
917     var isSet = SMUtils.definesProperty( args, 'value');
918     var jNode = $(args.node);
919     switch (jNode[0].nodeName.toLowerCase()) {
920         case "input": case "textarea":
921             if ( isSet ) { jNode.val( args.value); }
922                 else { return jNode.val(); }
923         break;
924 
925         case "abbr":
926             if ( isSet ) { jNode.attr( "title", args.value); }
927                 else { return jNode.attr( "title"); }
928         break;
929 
930         default:
931             if ( isSet ) { jNode.text( args.value); }
932                 else { return jNode.text(); }
933         break;
934     }
935 };

955  *  Split a Horn property path into tokens.
956  *  <p>
957  *  For example, <code>pathToTokens( "-a-0-b-2-2")</code> yields,
958  *  <code>[a, 0, b, 2, 2]</code>.
959  *

961  *

972 /**
973  *  Converts an Horn property path in internal form to its external,
974  *  JavaScript representation.
975  *  <p>
976  *  For example: <code>horn.toExternalPath( 'x-1-2-3-y-2-z') ===
977  *  'x[1][2][3].y[2].z'.</code>
978  *
979  *  @param {String} path internal Horn property path to convert
980  *
981  *  @public
982  */
983 Horn.toExternalPath = function( path ) {
984     return path.replace( /\-?(\d+)/g, "[$1]").replace( /\-/g, ".");
985 };

987 /**
988  *  Converts a property path in external JavaScript form to that used
989  *  internally.
990  *  <p>
991  *  For example: <code>horn.toInternalPath('x[1][2][3].y[2].z') ===
992  *  'x-1-2-3-y-2-z'</code>.
993  *  <p>
994  *  This function tolerates extraneous leading '-' characters.
995  *
996  *  @param {String} path the external property path to convert
997  *
998  *  @public
999  */
1000 Horn.toInternalPath = function( path ) {
1001     var rv = path.replace(
1002         /(\[(\w+)\])/g, ".$2").replace( /\./g, "-").replace( "/", "");
1003     return SMUtils.hasPrefix( rv, "-") ? rv.substring(1) : rv;
1004 };

1006 /**
1007  *  Converts a horn property path with optional wildcard '*' characters into
1008  *  a regular expression.
1009  *  <p>
1010  *  '*' wildcards are converted into '.*' regular expression terms. Array
1011  *  indexing and object dereference operators are correctly escaped.
1012  *
1013  *  @param {String} path the Horn property path to convert to a regular
1014  *      expression
1015  *
1016  *  @return {String} a horn property path regular expression
1017  *
1018  *  @public
1019  */
1020 Horn.toRegularExpression = function( path ) {
1021     return path.replace( /([\.\[\]])/g, "\\$1").replace( "*", ".*");

1024 /**
1025  *  Traverses an object graph and executes a callback function for each
1026  *  value encountered.
1027  *  <p>
1028  *  <code>Object</code> and <code>Array</code> types are considered
1029  *  containers of values rather than values themselves.
1030  *  <p>
1031  *  For any model value, the corresponding equivalent Horn property path is
1032  *  constructed and reported to the callback function. An optional property
1033  *  path stem may be supplied which will be prepended to any such path.
1034  *
1035  *  @param value the root of the object graph to traverse
1036  *  @param {Function} callback a callback function that will be executed for
1037  *      each value of the object graph encountered, it should have the
1038  *      following signature, <code>( path, value, context, propName)</code>,
1039  *      where 'path' is the value's property path within its container,
1040  *      'value' is the current value found, 'context' is the object in which
1041  *      the value can be found and 'key' is the property name within
1042  *      'context' that the value can be found.
1043  *  @param {String} [path] an optional property path that will be prefixed
1044  *      to every callback reported property path constructed
1045  *  @param [context] internal (recursive) use only - do not supply a value
1046  *  @param [propName] internal (recursive) use only  - do not supply a value
1047  *
1048  *  @public
1049  */
1050 Horn.traverse = function( value, callback, path, context, propName ) {
1051     if ( (value instanceof Object) || (value instanceof Array) ) {
1052         SMUtils.each( value, function( k, v ) { Horn.traverse( v, callback,
1053             path ? (path + '-' + k) : ("-" + k), value, k);
1054         }, this);
1055     } else { callback( path, value, context, propName); }
1056 };
