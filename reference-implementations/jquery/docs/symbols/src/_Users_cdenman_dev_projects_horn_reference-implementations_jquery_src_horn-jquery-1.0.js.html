 28     /**
 29      * @private
 30      * @function
 31      */
 32     var addBinding = SMUtils.bind( function( args ) {
 33         var rv;
 34         var details;
 35         if ( args.setModel !== false ) {
 36             details = setValue(  args.value, args.path);
 37         }
 38         if ( (args.readOnly === false) && (!args.isJSON) ) {
 39             if ( SMUtils.isDefinedNotNull( details) ) {
 40                 args.context = details.context;
 41                 args.key = details.key;
 42             }
 43             if ( !SMUtils.isDefinedNotNull( state.bindings) ) {
 44                 state.bindings = {}; }
 45             rv = {node: args.node, value: args.value};
 46             if ( SMUtils.isDefinedNotNull( args.context) ) {
 47                 rv.context = args.context;
 48                 rv.key = args.key;
 49             }
 50             state.bindings[ args.path] = rv;
 51         }
 52     }, this);
 53 
 54     /**
 55      * @private
 56      * @function
 57      */
 58     var addBindings = SMUtils.bind( function( args ) {
 59         SMUtils.each(
 60             args.bindings,
 61             function( i, newArgs ) {
 62                 var modelValue;
 63                 var textValue;
 64                 var ref = getModelReference( newArgs);
 65                 if ( SMUtils.isDefinedNotNull( ref) )  {
 66                     modelValue = ref.ref[ ref.key];
 67                     textValue = convert( {
 68                         value: modelValue,
 69                         type:  'toText',
 70                         path:  newArgs.path,
 71                         node:  newArgs.node
 72                     });
 73                     if ( (textValue === undefined) &&
 74                         SMUtils.isDefinedNotNull( modelValue)) {
 75                         textValue = modelValue + "";
 76                     }
 77                     newArgs.text = textValue;
 78                     this.hornNodeValue( {node: newArgs.node,
 79                         value: newArgs.text});
 80                     addBinding({
 81                         setModel: false,
 82                         value: modelValue,
 83                         node: newArgs.node,
 84                         key: ref.key,
 85                         context: ref.ref,
 86                         path: newArgs.path});
 87                 }
 88             }, this);
 89     }, this);

 91     /**
 92      * @private
 93      * @function
 94      */
 95     var addJSONBindings = SMUtils.bind(
 96         function( args ) {
 97             var defaults = {type:  'fromJSON', node:  args.node,
 98                 readOnly: args.readOnly};
 99             var addJSONHelper = SMUtils.bind( function( vargs ) {
100                 var oldValue = vargs.value;
101                 vargs.value = convert( vargs);
102                 if ( !SMUtils.isDefinedNotNull( vargs.value) ) {
103                     vargs.value = oldValue;
104                 }
105                 addBinding( vargs);
106             }, this);
107             var jsonData = $.evalJSON( args.text);
108             if ( typeof jsonData === 'object' ) {
109                 this.traverse( jsonData,
110                     SMUtils.bind( function( k, v ) {
111                         addJSONHelper( $.extend( defaults, { value: v,
112                             path:  args.path + k})); }, this));
113             } else {
114                 addJSONHelper( $.extend( defaults,
115                     {value: jsonData, path:  args.path}));
116             }
117         }, this);

119     /**
120      * @private
121      * @function
122      */
123     var convert = SMUtils.bind( function ( args ) {
124         var converter = state.opts.converter;
125         if ( !SMUtils.isDefinedNotNull( converter) ) { return undefined; }
126         return converter.call( this, $.extend( {}, args,
127             {path: this.toExternalPath( args.path)}))
128     }, this);

130     /**
131      * @private
132      * @function
133      */
134     var extract = SMUtils.bind( function( args ) {
135         var _this = this;
136         var pathStem = SMUtils.definesProperty( args, 'pathStem') ?
137             this.toInternalPath( args.pathStem) : undefined;
138         var rootNodes = SMUtils.definesProperty( args, 'nodes') ?
139             $(args.nodes) : this.rootNodes();
140         setDefaultModel();
141         SMUtils.each( rootNodes,
142             function( i, n ) {
143                 var inGraph = false;
144                 this.walkDOM( n,
145                     function( n, path ) {
146                         if ( _this.hasRootIndicator(n) ) {
147                             if ( inGraph ) { return false; }
148                                 else { inGraph = true; }
149                         }
150                         var bindingData = _this.hasHornBinding( n);
151                         if ( bindingData === false ) {
152                             return true; }
153                         bindingData.readOnly = args.readOnly;
154                         bindingData.path = _this.combinePaths( pathStem, path);
155                         if ( bindingData.isJSON === false ) {
156                             bindingData.value = convert( {
157                                 value: bindingData.text,
158                                 path:  bindingData.path,
159                                 type:  'fromText',
160                                 node:  bindingData.node
161                             });
162                             if ( !SMUtils.isDefinedNotNull( bindingData.value) ) {
163                                 bindingData.value = bindingData.text;
164                             }
165                             addBinding( bindingData);
166                         } else {
167                             addJSONBindings( bindingData);
168                         }
169                         return false;
170                     }
171                 );
172             }, this);
173         return state.model;
174     }, this);

176     /**
177      * @private
178      * @function
179      */
180     var getModelReference = SMUtils.bind( function( args ) {
181         var rv;
182         var tokens = this.pathToTokens( args.path);
183         var length = tokens.length;
184         if ( length > 0 ) {
185             rv = {ref: state.model, key: tokens[ length - 1]};
186             tokens.length = tokens.length - 1;
187             SMUtils.each( tokens, function( i, n ) {
188                 if ( SMUtils.isDefinedNotNull( rv.ref) ) {
189                     if ( rv.ref.hasOwnProperty( n) ) { rv.ref = rv.ref[ n]; }
190                 } else { return false; }
191             }, this);
192             if ( !SMUtils.isDefinedNotNull( rv.ref) ) { rv = undefined; }
193         }
194         return rv;
195     }, this);

197     /**
198      *  @private
199      *  @function
200      */
201     var handleTemplateBinding = SMUtils.bind( function( node, path, bindings ) {
202         var bindingData = this.hasHornBinding( node);
203         if ( (bindingData !== false) && !bindingData.isJSON ) {
204             bindingData.path = path;
205             bindings.push( bindingData);
206             return false;
207         }
208         return true;
209     }, this);

211     /**
212      * @private
213      * @function
214      */
215     var render = SMUtils.bind( function( args ) {
216         var rootNode = args.rootNode;
217         var binding = args.binding;
218         var modelValue = binding.context[ binding.key];
219         var textValue;
220         var cArgs;
221         if ( modelValue !== binding.value ) {
222             if ( !rootNode || (rootNode && SMUtils.contains(
223                 $(binding.node).parents(), rootNode)) ) {
224                 cArgs = {
225                     value: modelValue,
226                     path:  args.path,
227                     type:  'toText',
228                     node: binding.node};
229                 textValue = convert( cArgs);
230                 if ( !SMUtils.isDefinedNotNull( textValue) ) {
231                     textValue = modelValue + "";
232                 }
233                 this.hornNodeValue( {node: binding.node, value: textValue});
234                 binding.value = modelValue;
235                 return binding.node;
236             }
237         }
238     }, this);

244     var setDefaultArgs = SMUtils.bind( function( args ) {
245         var existingArgs = SMUtils.definesProperty(
246             args, 'args') ? args.args : {};
247         if ( SMUtils.definesProperty( args, 'defaults') ) {
248             $.extend( existingArgs, args.defaults);
249         }
250         return existingArgs;

258         if ( !SMUtils.isDefinedNotNull( state.model)
259             && state.opts.hasOwnProperty( 'defaultModel') ) {
260             state.model = state.opts.defaultModel; // @todo clone here
261         }

264     /**
265      * @private
266      * @function
267      */
268     var setValue = SMUtils.bind( function( value, path, parentContext ) {
269         var token;
270         var numTokens;
271         var subContext;
272         if ( typeof path === 'string' ) {
273             path = this.pathToTokens( path);
274             if ( !SMUtils.isDefinedNotNull( state.model) ) {
275                 state.model = (!isNaN( parseInt( path[ 0])) ? [] : {});
276             }
277             parentContext = state.model;
278         }
279         numTokens = path.length;
280         if ( numTokens > 0 ) {
281             token = path.shift();
282             if ( numTokens > 1 ) {
283                 if ( !parentContext.hasOwnProperty( token) ) {
284                     subContext = !isNaN( parseInt( path[ 0])) ? [] : {};
285                     parentContext[ token] = subContext;
286                 }
287                 subContext = parentContext[ token];
288                 return setValue( value, path, subContext);
289             } else {
290                 parentContext[ token] = value;
291                 return {context: parentContext, key: token, value: value};
292             }
293         }
294     }, this);

296     /**
297      *  Walk DOM tree(s) and extract model data, allowing for subsequent model
298      *  to DOM updates.
299      *  <p>
300      *  After execution, each Horn value element encountered will have a
301      *  corresponding representation in the model. Altering such model
302      *  values and then calling <code>updateDOM(...)</code> will refresh their
303      *  display values.
304      *  <p>
305      *  The DOM tree(s) to walk may be specified in exactly one of two ways:
306      *  <ol>
307      *      <li>
308      *          Horn will automatically find all DOM trees that have a root
309      *          indicator.
310      *      </li>
311      *      <li>
312      *          Passing in 'args.nodes', a collection of DOM elements or a
313      *          jQuery node selector.
314      *      </li>
315      *  </ol>
316      *  <p>
317      *  Before a value is stored in the model it is converted to its
318      *  <code>String</code> representation. Alternatively, applications can
319      *  register converter functions to override this default behaviour.
320      *  <p>
321      *  If model to DOM updates are not required, the alternative yet otherwise
322      *  identical function, <code>{@link Horn#load}</code> should be used
323      *  (or the 'readOnly' option).
324      *
325      *  @param {Object} args
326      *  @param {Element|String} [args.nodes] DOM nodes or a jQuery selector
327      *  @param {String} [args.pathStem] a property path prefix that will be
328      *      appended the paths of value nodes' property paths before storage in
329      *      the model
330      *
331      *  @return the updated model
332      *
333      *  @see Horn#option for the detailing of converter functions
334      *  @see Horn#load
335      *
336      *  @public
337      */
338     this.bind = function( args ) {
339         return extract(
340             setDefaultArgs( {args: args, defaults: {readOnly: false}}));
341     };

343     /**
344      *  Returns all <strong>bound</strong> model-entry paths that contain blank
345      *  <code>""</code> <code>String</code> values.
346      *  <p>
347      *  The model paths can be optionally filtered by requiring them to have a
348      *  given prefix.
349      *  <p>
350      *  Also optionally, the DOM nodes corresponding to the given model entry
351      *  can be content-checked to determine if it (instead) has a blank
352      *  <code>String</code> value.
353      *
354      *  @param {Object} args
355      *  @param {String} [args.path] the prefix that returned property paths must
356      *      have, if not supplied all property paths will be returned - if they
357      *      fulfill the content requirements
358      *  @param {Boolean} [args.inspectDOM] iff. <code>true<code> then instead
359      *      of using the model entries' value to compare with <code>""</code>,
360      *      the model entries' corresponding DOM node's value will be used
361      *      instead
362      *
363      *  @return {Array} an <code>Array</code> containing all model entries that
364      *      have the given (or any) path prefix and have either their model, or
365      *      DOM node values equal to the blank <code>String</code>
366      *      <code>""</code>
367      *
368      *  @public
369      */
370     this.blankModelEntries = function( args ) {
371         var blankPaths = [];
372         var pathDefined = SMUtils.definesProperty( args, 'path');
373         var inspectDOM = SMUtils.definesProperty( args, 'inspectDOM') &&
374             (args.inspectDOM === true);
375         var path = pathDefined ? this.toInternalPath( args.path) : undefined;
376         SMUtils.each( state.bindings, function( i, n ) {
377             if ( (!pathDefined || SMUtils.hasPrefix( i, path)) ) {
378                 if (((inspectDOM && SMUtils.isDefinedNotNull( n.node)) ?
379                     this.hornNodeValue( {node: n.node}) : n.value) === "") {
380                     blankPaths.push(this.toExternalPath(i));
381                 }
382             }
383         }, this);
384         return blankPaths;
385     };

387     /**
388      *  Walks a tree of cloned (or selected) DOM nodes, binding them to model
389      *  elements and populating their display values.
390      *  <p>
391      *  During the walk, the property path for each DOM value node visited is
392      *  inspected and if there is a corresponding model element with the same
393      *  path it will be bound to the DOM node and the DOM node will receive its
394      *  value.
395      *  <p>
396      *  For each DOM value node encountered, the optional 'args.pathStem' will
397      *  be added to its property path before binding to a model value. Allowing
398      *  nodes derived from a single template to be bound to multiple model
399      *  locations.
400      *  <p>
401      *  Model to DOM data conversions are performed using this function with the
402      *  type parameter value 'toText'.
403      *
404      *  @param {Object} args
405      *  @param {Element|String} [args.node] a jQuery node or selector - nodes
406      *      selected by this argument will not be cloned.
407      *  @param {Element|String} [args.template] jQuery node or selector - these
408      *      nodes will be cloned and the root node will have its 'id' attribute
409      *      removed and set to 'args.newID' if supplied
410      *  @param {String} [args.id] the new 'id' attribute value applied to nodes
411      *      cloned, if using 'args.template'
412      *  @param {String} [args.pathStem] a property path prefix appended to all
413      *      walked DOM nodes before binding to model values
414      *
415      *  @return the newly cloned and populated template
416      *
417      *  @public
418      */
419     this.bindTo = function( args ) {
420         var node;
421         var pathStem;
422         var bindings = [];
423         if ( SMUtils.definesProperty( args, 'node') ) {
424             node = $(args.node);
425         } else {
426             node = $(args.template).clone();
427             node.removeAttr( "id");
428             if ( SMUtils.definesProperty( args, 'id') ) {
429                node.attr( "id", args.id);
430             }
431         }
432         setDefaultModel();
433         pathStem = SMUtils.definesProperty( args, 'pathStem') ?
434             this.toInternalPath( args.pathStem) : '';
435         this.walkDOM( node,
436             SMUtils.bind( function( n, path ) {
437                 return handleTemplateBinding( n, path, bindings);
438             }, this), pathStem);
439         addBindings({bindings: bindings});
440         return node;
441     };

491     /**
492      *  Get an option's value by name, or set an option's value by name.
493      *  <p>
494      *  If no value is provided, the value of the named option is returned,
495      *  otherwise the return value is undefined.
496      *  <p>
497      *  The following options are currently supported:<br>
498      *  <ul>
499      *      <li>
500      *          <p>
501      *          <strong>defaultModel</strong> - For setting an explicit default
502      *          model (<code>Object</code> or <code>Array</code>)
503      *          </p>
504      *      </li>
505      *      <li>
506      *          <p>
507      *          <strong>readOnly</strong> - If set to true, the automatic
508      *          extraction at startup will call load() instead of bind() so
509      *          there is no two-way binding to the DOM.
510      *          </p>
511      *      </li>
512      *      <li>
513      *          <p>
514      *          <strong>converter</strong> - A function with the following
515      *          signature, <code>(args)</code> that will be called upon to
516      *          convert values under the following scenarios:
517      *          </p>
518      *          <p>
519      *              <ul>
520      *                  <li>
521      *                      <p>
522      *                      <strong>DOM to model</strong> :- when values are
523      *                      extracted from the DOM for storage in the model
524      *                      (functions <code>{@link Horn#bind}</code>,
525      *                      <code>{@link Horn#load}</code>) the
526      *                      <code>convert</code> function
527      *                      will be called with 'type' parameter value
528      *                      '<strong>fromText</strong>'.
529      *                      </p>
530      *                  </li>
531      *                  <li>
532      *                      <p>
533      *                      <strong>Model to DOM</strong> :- when values
534      *                      read from the model are destined for the DOM,
535      *                      (functions <code>{@link Horn#updateDOM}</code>,
536      *                      <code>{@link Horn#bindTo}</code>) the
537      *                      <code>convert</code> function will be called with
538      *                      'type' parameter value '<strong>toText</strong>'.
539      *                      </p>
540      *                  </li>
541      *                  <li>
542      *                      <p>
543      *                      <strong>JSON to model</strong> :- when JSON values
544      *                      are extracted from the DOM for model storage
545      *                      (functions <code>{@link Horn#bind}</code>,
546      *                      <code>{@link Horn#load}</code>), the
547      *                      <code>convert</code> function will be called with
548      *                      'type' parameter value '<strong>fromJSON</strong>'.
549      *                      </p>
550      *                  </li>
551      *              </ul>
552      *          </p>
553      *          <p>
554      *              The complete set of arguments passed to the
555      *              <code>convert</code> function are namely:
556      *              <ul>
557      *                  <li>
558      *                      <p>
559      *                      <strong>args.type</strong> :- the conversion 'type'
560      *                      as described above.
561      *                      </p>
562      *                  </li>
563      *
564      *                  <li>
565      *                      <p>
566      *                      <strong>args.value</strong> :- the value for
567      *                      conversion. The JavaScript type is dependent on the
568      *                      conversion being performed (<code>String</code> for
569      *                      '<strong>fromText</strong>' conversions and the
570      *                      corresponding model value's type in all others).
571      *                      </p>
572      *                  </li>
573      *
574      *                  <li>
575      *                      <p>
576      *                      <strong>args.path</strong> :- the property path for
577      *                      the DOM or model value.
578      *                      </p>
579      *                  </li>
580      *
581      *                  <li>
582      *                      <p>
583      *                      <strong>args.node</strong> :- the DOM node that
584      *                      generated the value.
585      *                      </p>
586      *                  </li>
587      *              </ul>
588      *          </p>
589      *          <p>
590      *              Converter functions can return <code>undefined</code> in
591      *              which case the value (all such values) will be treated as
592      *              <code>String</code> types - the same behaviour as having no
593      *              converter function registered.
594      *          </p>
595      *          <p>
596      *              <strong>Only a single converter function can be registered,
597      *              with any invocation overwriting any previously registered
598      *              function.</strong>
599      *          </p>
600      *      </li>
601      *  </ul>
602      *
603      *  @param {Object} optionName the name of the option to set
604      *  @param {Object} [value] the value to set
605      *
606      *  @return the value of the named option if no value supplied else
607      *      <code>undefined</code>
608      *
609      *  @public
610      */
611     this.option = function( optionName, value ) {
612         if ( value !== undefined ) {
613             state.opts[ optionName] = value;
614         } else { return state.opts[ optionName]; }
615     };

631     /**
632      *  Removes either, all bindings or, all bindings with a given property path
633      *  , or all bindings with property path's that match a given regular
634      *  expression.
635      *  <p>
636      *  If no arguments are defined, <strong>all bindings are removed</strong>.
637      *  <p>
638      *  If both 'args.path' and 'args.pattern' are supplied, the result is the
639      *  same as calling this function twice, once supplying 'args.path' and
640      *  thence 'args.pattern'.
641      *
642      *  @param {Object} args
643      *  @param {String} [args.path] all bindings with property path's that
644      *      equal this argument, will be removed
645      *  @param {String} [args.pattern] all bindings with property path's that
646      *      match this regular expression, will be removed
647      *
648      *  @public
649      */
650     this.unbind = function( args ) {
651         var definesPath = SMUtils.definesProperty( args, 'path');
652         var internalPath = definesPath ? this.toInternalPath( args.path) :
653             undefined;
654         var definesPattern = SMUtils.definesProperty( args, 'pattern');
655         var unbindAll = !definesPattern && !definesPath;
656         SMUtils.each( state.bindings,
657             function( i, n ) {
658                 if (    unbindAll ||
659                         (definesPath && (i === internalPath)) ||
660                         (definesPattern && (i.match( args.pattern)))) {
661                     delete state.bindings[ i];
662                 }
663             }, this);
664     };

666     /**
667      *  Update all bound DOM nodes with their current model values, if altered.
668      *  <p>
669      *  This function will not update DOM nodes if their model value has not
670      *  changed.
671      *
672      *  @param {Element} rootNode optional DOM node such that if supplied, only
673      *      nodes under this nodes will be updated
674      *
675      *  @return {Array} an array of nodes that had their DOM values changed
676      *
677      *  @public
678      */
679     this.updateDOM = function( rootNode ) {
680         var alteredNodes = [];
681         SMUtils.each( state.bindings, function( i, n ) {
682             var node = render( {rootNode: rootNode, binding: n, path: i});
683             if ( SMUtils.isDefinedNotNull( node) ) { alteredNodes.push( node); }
684         }, this);
685         return alteredNodes;
686     };

688     this.reset();

691 Horn.prototype = {
692 
693     /**
694      *  Join parent and child (internal) property paths.
695      *  <p>
696      *  If exactly one path is <code>null</code> or <code>undefined</code>, the
697      *  other is returned. If both paths are not defined, the empty
698      *  <code>String</code> is returned.
699      *
700      *  @param {String} [parent] the parent horn property path
701      *  @param {String} [child] the child horn property path
702      *
703      *  @return {String} the resultant, combined property path
704      *
705      *  @methodOf Horn.prototype
706      */
707     combinePaths: function( parent, child ) {
708         var parentDefined = this.isPathDefined( parent);
709         var childDefined = this.isPathDefined( child);
710         if ( parentDefined && childDefined ) {
711             return parent + "-" + child;
712         } else if ( parentDefined ) {
713             return parent;
714         } else if ( childDefined ) {
715             return child;
716         } else { return ""; }
717     },
718 
719     /**
720      *  Determines if a given node in the context of a Horn DOM tree is a value
721      *  node or not.
722      *  <p>
723      *  If the node is a value node, the necessary binding information is
724      *  extracted and returned.
725      *
726      *  @param node the DOM node to examine
727      *
728      *  @return <code>false</code> if the given node is not a value node else,
729      *      this function returns an object containing the binding information
730      *      extracted
731      *
732      *  @methodOf Horn.prototype
733      */
734     hasHornBinding: function( node ) {
735         var theContained;
736         var nodeName;
737         var contents = $($(node).contents());
738         var isPathDefined = this.isPathDefined( this.pathIndicator(node));
739         var cd = { isJSON: this.hasJSONIndicator(node), node: node};
740         var contentsSize = contents.size();
741         var isEmptyNode = contentsSize === 0;
742         if ( (contentsSize === 1) || (isEmptyNode && !cd.isJSON))  {
743             if ( !isEmptyNode ) { theContained = contents[0]; }
744             if ( cd.isJSON || isPathDefined ) {
745                 nodeName = node.nodeName.toLowerCase();
746                 cd.isFormElementNode =
747                     (nodeName === 'input') || (nodeName === 'textarea');
748                 cd.isABBRNode = !cd.isFormElementNode &&
749                     (nodeName.toLowerCase() === "abbr");
750                 cd.isTextNode = !cd.isABBRNode && (isEmptyNode ||
751                     (theContained.nodeType === Node.TEXT_NODE));
752                 if ( cd.isFormElementNode || cd.isTextNode || cd.isABBRNode ) {
753                     cd.text = this.hornNodeValue( {node: node});
754                     return cd;
755                 }
756             }
757         }
758         return false;
759     },
760 
761     /**
762      *  Sets or retrieves a DOM node's Horn text.
763      *  <p>
764      *  The value retrieved is HTML un-escaped.
765      *
766      *  @param {Object} args
767      *  @param {Element} args.node the node from which to retrieve text
768      *  @param {Object} args.value the value to set
769      *
770      *  @return {String} the given node's displayed text
771      *
772      *  @methodOf Horn.prototype
773      */
774     hornNodeValue: function( args ) {
775         var isSet = SMUtils.definesProperty( args, 'value');
776         var jNode = $(args.node);
777         switch (jNode[0].nodeName.toLowerCase()) {
778             case "input": case "textarea":
779                 if ( isSet ) { jNode.val( args.value); }
780                     else { return jNode.val(); }
781             break;
782 
783             case "abbr":
784                 if ( isSet ) { jNode.attr( "title", args.value); }
785                     else { return jNode.attr( "title"); }
786             break;
787 
788             default:
789                 if ( isSet ) { jNode.text( args.value); }
790                     else { return jNode.text(); }
791             break;
792         }
793     },
794 
795     /**
796      *  Determines if a given value represents a property path.
797      *  <p>
798      *  The 'path' argument is converted to a <code>String</code> before
799      *  examination.
800      *
801      *  @param {Object} path an object that represents a property path
802      *
803      *  @return {Boolean} <code>true</code> if the given path is a defined
804      *      property path, <code>false</code> otherwise
805      *
806      *  @methodOf Horn.prototype
807      */
808     isPathDefined: function ( path ) {
809         return SMUtils.isDefinedNotNull( path) && ((path + "").trim() !== '');
810     },
811 
812     /**
813      *  Split a Horn property path into tokens.
814      *  <p>
815      *  For example, <code>pathToTokens( "-a-0-b-2-2")</code> yields,
816      *  <code>[a, 0, b, 2, 2]</code>.
817      *
818      *  @param {String} path the Horn property path to split
819      *
820      *  @return {Array} the tokens extracted from the property path
821      *
822      *  @methodOf Horn.prototype
823      */
824     pathToTokens: function( path ) {
825         return path ? ((SMUtils.hasPrefix( path, "-") ||
826             SMUtils.hasPrefix( path, "_"))  ?
827                 path.substring( 1) : path).split( "-") : undefined;
828     },
829 
830     /**
831      *  Converts an Horn property path in internal form to its external,
832      *  JavaScript representation.
833      *  <p>
834      *  For example: <code>horn.toExternalPath( 'x-1-2-3-y-2-z') ===
835      *  'x[1][2][3].y[2].z'.</code>
836      *
837      *  @param {String} path internal Horn property path to convert
838      *
839      *  @methodOf Horn.prototype
840      */
841     toExternalPath: function( path ) {
842         return path.replace( /\-?(\d+)/g, "[$1]").replace( /\-/g, ".");
843     },
844 
845     /**
846      *  Converts a property path in external JavaScript form to that used
847      *  internally.
848      *  <p>
849      *  For example: <code>horn.toInternalPath('x[1][2][3].y[2].z') ===
850      *  'x-1-2-3-y-2-z'</code>.
851      *  <p>
852      *  This function tolerates extraneous leading '-' characters.
853      *
854      *  @param {String} path the external property path to convert
855      *
856      *  @methodOf Horn.prototype
857      */
858     toInternalPath: function( path ) {
859         var rv = path.replace(
860             /(\[(\w+)\])/g, ".$2").replace( /\./g, "-").replace( "/", "");
861         return SMUtils.hasPrefix( rv, "-") ? rv.substring(1) : rv;
862     },
863 
864     /**
865      *  Traverses an object graph and executes a callback function for each
866      *  value encountered.
867      *  <p>
868      *  <code>Object</code> and <code>Array</code> types are considered
869      *  containers of values rather than values themselves.
870      *  <p>
871      *  For any model value, the corresponding equivalent Horn property path is
872      *  constructed and reported to the callback function. An optional property
873      *  path stem may be supplied which will be prepended to any such path.
874      *
875      *  @param value the root of the object graph to traverse
876      *  @param {Function} callback a callback function that will be executed for
877      *      each value of the object graph encountered, it should have the
878      *      following signature, <code>( path, value, context, propName)</code>,
879      *      where 'path' is the value's property path within its container,
880      *      'value' is the current value found, 'context' is the object in which
881      *      the value can be found and 'key' is the property name within
882      *      'context' that the value can be found.
883      *  @param {String} [path] an optional property path that will be prefixed
884      *      to every callback reported property path constructed
885      *  @param [context] internal (recursive) use only - do not supply a value
886      *  @param [propName] internal (recursive) use only  - do not supply a value
887      *
888      *  @methodOf Horn.prototype
889      */
890     traverse: function( value, callback, path, context, propName ) {
891         if ( (value instanceof Object) || (value instanceof Array) ) {
892             SMUtils.each( value, function( k, v ) { this.traverse( v, callback,
893                 path ? (path + '-' + k) : ("-" + k), value, k);
894             }, this);
895         } else { callback( path, value, context, propName); }
896     },
897 
898     /**
899      *  Walks the DOM and executes a callback for each node visited, building
900      *  up a property paths as it goes.
901      *  <p>
902      *  This function takes a property path stem it prepends to all Horn paths
903      *  constructed.
904      *  <p>
905      *  The callback function can stop at a given node by returning a non
906      *  <code>true</code> value.
907      *
908      *  @param {Element} node the node to start the walk from, this node is
909      *      implicitly visited (the callback will not be executed in respect
910      *      of it)
911      *  @param {Function} callback the callback function with the following
912      *      signature (node, path)  - where node is the current node being
913      *      visited and path is its full property path (relative to the first
914      *      DOM node visited and the path argument to <code>walkDOM(...)</code>
915      *      proper
916      *  @param {String} [path] the Horn property path stem that will be
917      *      prepended to each Horn path constructed
918      *
919      *  @methodOf Horn.prototype
920      */
921     walkDOM: function( node, callback, path ) {
922         if ( !SMUtils.isDefinedNotNull( path) ) { path = ''; }
923         path = this.combinePaths( path, this.pathIndicator(node));
924         if ( callback( node, path) === true ) {
925             SMUtils.each( $(node).children(), function( i, n ) {
926                 this.walkDOM( n, callback, path); }, this);
927         }
928     }
