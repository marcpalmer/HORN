 27     /**
 28      * @private
 29      * @function
 30      */
 31     var addBinding = this.scope( function( args ) {
 32         var details;
 33         var culledPath = args.path.substring( 1);
 34         var rv;
 35         if ( args.setModel !== false ) {
 36             details = setValue(  args.value, args.path);
 37         }
 38         if ( (args.readOnly === false) && (!args.isJSON) ) {
 39             if ( this.isDefinedNotNull( details) ) {
 40                 args.context = details.context;
 41                 args.key = details.key;
 42             }
 43 
 44             if ( !this.isDefinedNotNull( state.bindings) ) {
 45                 state.bindings = {}; }
 46             rv = {node: args.node, value: args.value};
 47             if ( this.isDefinedNotNull( args.context) ) {
 48                 rv.context = args.context;
 49                 rv.key = args.key;
 50             }
 51             state.bindings[ culledPath] = rv;
 52         }
 53     }, this);
 54 
 55     /**
 56      * @private
 57      * @function
 58      */
 59     var addBindings = this.scope( function( args ) {
 60         this.each(
 61             args.bindings,
 62             function( i, newArgs ) {
 63                 var modelValue;
 64                 var textValue;
 65                 var ref = getModelReference( newArgs);
 66                 if ( this.isDefinedNotNull( ref) )  {
 67                     modelValue = ref.ref[ ref.key];
 68                     textValue = convert( {
 69                         value: modelValue,
 70                         path:  newArgs.path,
 71                         type:  'toText',
 72                         node:  newArgs.node
 73                     });
 74                     if ( (textValue === undefined) &&
 75                         this.isDefinedNotNull( modelValue)) {
 76                         textValue = modelValue + "";
 77                     }
 78                     newArgs.text = textValue;
 79                     this.setHornDOMNodeValue( {node: newArgs.node,
 80                         value: newArgs.text});
 81                     addBinding({
 82                         setModel: false,
 83                         node: newArgs.node,
 84                         context: ref.ref,
 85                         key: ref.key,
 86                         value: modelValue,
 87                         path: newArgs.path});
 88                 }
 89             }, this);
 90     }, this);

 92     /**
 93      * @private
 94      * @function
 95      */
 96     var addJSONBindings = this.scope(
 97         function( args ) {
 98             var defaults = {type:  'fromJSON', node:  args.node,
 99                 readOnly: args.readOnly};
100             var addJSONHelper = this.scope( function( vargs ) {
101                 var oldValue = vargs.value;
102                 vargs.value = convert( vargs);
103                 if ( !this.isDefinedNotNull( vargs.value) ) {
104                     vargs.value = oldValue;
105                 }
106                 addBinding( vargs);
107             }, this);
108             var jsonData = $.evalJSON( args.text);
109             var rootPath = args.path;
110             if ( typeof jsonData === 'object' ) {
111                 this.traverse( jsonData,
112                     this.scope( function( k, v ) {
113                         addJSONHelper( $.extend( defaults,
114                             {value: v, path:  rootPath + k}));
115                     }, this));
116             } else {
117                 addJSONHelper( $.extend( defaults,
118                     {value: jsonData, path:  rootPath}));
119             }
120         }, this);

122     /**
123      * @private
124      * @function
125      */
126     var convert = this.scope( function ( args ) {
127         var converter = state.opts.converter;
128         if ( this.startsWith( args.path, "-") ) {
129             args.path = args.path.substring( 1);
130         }
131         return this.isDefinedNotNull( converter) ?
132             converter.call( this, args) : undefined;
133     }, this);

135     /**
136      * @private
137      * @function
138      */
139     var extract = this.scope( function( args ) {
140         var _this = this;
141         var rootNodes = this.definesProperty( args, 'rootNodes') ?
142             args.rootNodes : (this.definesProperty( args, 'selector') ?
143                 $(args.selector) : this.rootNodes());
144         setDefaultModel();
145         this.each( rootNodes,
146             function( i, n ) { this.visitNodes( n, '',
147                 function( n, path ) {
148                     if ( _this.hasRootIndicator( { n: n}) ) {
149                         return false; }
150                     var bindingData = _this.hasHornBinding( n, path);
151                     if ( bindingData === false ) {
152                         return true; }
153                     bindingData.readOnly = args.readOnly;
154                     if ( bindingData.isJSON === false ) {
155                         bindingData.value = convert( {
156                             value: bindingData.text,
157                             path:  bindingData.path,
158                             type:  'fromText',
159                             node:  bindingData.node
160                         });
161                         if ( !_this.isDefinedNotNull( bindingData.value) ) {
162                             bindingData.value = bindingData.text;
163                         }
164                         addBinding( bindingData);
165                     } else {
166                         addJSONBindings( bindingData);
167                     }
168                     return false;
169                 }
170             ); }, this);
171         return state.model;
172     }, this);

174     /**
175      * @private
176      * @function
177      */
178     var getModelReference = this.scope( function( args ) {
179         var rv;
180         var tokens = this.pathToTokens( args.path);
181         var length = tokens.length;
182         if ( length > 0 ) {
183             rv = {ref: state.model, key: tokens[ length - 1]};
184             tokens.length = tokens.length - 1;
185             this.each( tokens, function( i, n ) {
186                 if ( this.isDefinedNotNull( rv.ref) ) {
187                     if ( rv.ref.hasOwnProperty( n) ) { rv.ref = rv.ref[ n]; }
188                 } else { return false; }
189             }, this);
190             if ( !this.isDefinedNotNull( rv.ref) ) { rv = undefined; }
191         }
192         return rv;
193     }, this);
194 
195     /**
196      *  @private
197      *  @function
198      */
199     var handleTemplateBinding = this.scope( function( node, path, bindings ) {
200         var key;
201         var bindingData = this.hasHornBinding( node, path);
202         if ( bindingData !== false ) {
203             key = this.pathIndicator({n: node});
204             if ( this.isAdjustingPath( key) ) {
205             bindingData.path = (path + '-' + key); }
206             if ( !bindingData.isJSON  ) {
207             bindings.push( bindingData); }
208             return false;
209         }
210         return true;
211     }, this);

213     /**
214      * @private
215      * @function
216      */
217     var render = this.scope( function( args ) {
218         var rootNode = args.rootNode;
219         var binding = args.binding;
220         var modelValue = binding.context[ binding.key];
221         var textValue;
222         if ( modelValue !== binding.value ) {
223             if ( !rootNode || (rootNode &&
224                 this.contains( $(binding.node).parents(), rootNode)) ) {
225                 textValue = convert( {
226                     value: modelValue,
227                     path:  args.path,
228                     type:  'toText',
229                     node: binding.node});
230                 if ( !this.isDefinedNotNull( textValue) ) {
231                     textValue = modelValue + "";
232                 }
233                 this.setHornDOMNodeValue( {node: binding.node, value: textValue});
234                 binding.value = modelValue;
235                 return binding.node;
236             }
237         }
238     }, this);

245         var existingArgs = !this.definesProperty( args, 'args') ? {} : args.args;

247             $.extend( existingArgs, args.defaults);
248         }
249         return existingArgs;

252     /**
253      * @private
254      * @function

256     var setDefaultModel = this.scope( function() {
257         if ( !this.isDefinedNotNull( state.model)
258             && state.opts.hasOwnProperty( 'defaultModel') ) {
259             state.model = state.opts.defaultModel;
260         }

263     /**
264      * @private
265      * @function
266      */
267     var setValue = this.scope( function( value, path, parentContext ) {
268         var token;
269         var numTokens;
270         var subContext;
271         if ( typeof path === 'string' ) {
272             path = this.pathToTokens( path);
273             if ( !this.isDefinedNotNull( state.model) ) {
274                 state.model = (!isNaN( parseInt( path[ 0])) ? [] : {});
275             }
276             parentContext = state.model;
277         }
278         numTokens = path.length;
279         if ( numTokens > 0 ) {
280             token = path.shift();
281             if ( numTokens > 1 ) {
282                 if ( !parentContext.hasOwnProperty( token) ) {
283                     subContext = !isNaN( parseInt( path[ 0])) ? [] : {};
284                     parentContext[ token] = subContext;
285                 }
286                 subContext = parentContext[ token];
287                 return setValue( value, path, subContext);
288             } else {
289                 parentContext[ token] = value;
290                 return {context: parentContext, key: token, value: value};
291             }
292         }
293     }, this);

295     /**
296      *  Walk DOM tree(s) and extract model data, allowing for subsequent
297      *  updating.
298      *  <p>
299      *  After execution, each value element encountered will have a
300      *  corresponding representation in the model. Altering such model
301      *  values and then calling <code>updateDOM(...)</code> will refresh their
302      *  displayed value.
303      *  <p>
304      *  The DOM tree(s) to walk can be specified in exactly one of three ways:
305      *  <ol>
306      *      <li>
307      *          Horn will automatically find all DOM trees that have a root
308      *          indicator.
309      *      </li>
310      *      <li>
311      *          Passing a collection of DOM elements as
312      *          <code>args.rootNodes</code>.
313      *      </li>
314      *      <li>
315      *          Passing a JQuery selector string as <code>args.selector.</code>
316      *      </li>
317      *  </ol>
318      *  <p>
319      *  Before a value is stored in the model it is converted to its
320      *  <code>String</code> representation. Alternatively, applications can
321      *  register converter functions to override this default behaviour.
322      *  <p>
323      *  If later, model to DOM updates are not required, the alternative yet
324      *  otherwise identical function, <code>{@link Horn#load}</code> should be
325      *  used.
326      *
327      *  @param {Object} [rootNodes] a collection of DOM nodes to bind from, overrides
328      *      the default node selection mechanism
329      *  @param {String} [selector] a jQuery DOM node selector for nodes to bind
330      *      from,
331      *
332      *  @return the updated model
333      *
334      *  @see Horn#option for the detailing of converter functions
335      *  @see Horn#load
336      *
337      *  @public
338      */
339     this.bind = function( args ) {
340         return extract(
341             setDefaultArgs( {args: args, defaults: {readOnly: false}}));
342     };

344     /**
345      *  Clone an HTML template, then walk it, binding values encountered.  
346      *  <p>
347      *  
348      *
349      *  Create a new UI element by cloning an existing template that is
350      *  marked up with Horn indicators, and populate the DOM nodes with
351      *  data from the specified property path.
352      *
353      *  @param {String} args.path the property path within the model, to use to populate
354      *      this DOM node and its descendants
355      *
356      *  @param {Object} [args.data]
357      *  @param {Element|String} [args.node] a jQuery node or selector String
358      *  @param {Element|String} [args.template] jQuery node or selector String
359      *  @param [args.id] the new 'id' attribute value for a cloned args.template
360      *
361      *  @return the newly cloned and populated template
362      *
363      *  @public
364      */
365     this.bindTo = function( args ) {
366         var node;
367         var pathStem;
368         var bindings = [];
369         if ( this.definesProperty( args, 'node') ) {
370             node = $(args.node);
371         } else {
372             node = $(args.template).clone();
373             node.removeAttr( "id");
374             if ( this.definesProperty( args, 'id') ) {
375                node.attr( "id", args.id);
376             }
377         }
378 
379         setDefaultModel();
380 
381         pathStem = this.definesProperty( args, 'path') ? args.path : '';
382 
383         this.visitNodes( node, pathStem,
384             this.scope( function( n, path ) {
385                 return handleTemplateBinding( n, path, bindings);
386             }, this));
387 
388         addBindings({bindings: bindings});
389 
390         return node;
391     };

418     /**
419      *  Get an option's value by name, or set an option's value by name.
420      *  <p>
421      *  If no value is provided, the value of the named option is returned,
422      *  otherwise the return value is undefined.
423      *  <p>
424      *  The following options are currently supported:<br>
425      *  <ul>
426      *      <li><strong>defaultModel</strong> - for setting an explicit default
427      *          model (<code>Object</code> or <code>Array</code>)</li>
428      *      <li><strong>readOnly</strong> - </li>
429      *      <li><strong>converter</strong> - </li>
430      *  </ul>
431      *
432      *  @param {Object} optionName the name of the option to set
433      *  @param {Object} [value] the value to set
434      *
435      *  @return the value of the named option if no value supplied else
436      *      <code>undefined</code>
437      *
438      *  @public
439      */
440     this.option = function( optionName, value ) {
441         if ( value !== undefined ) {
442             state.opts[ optionName] = value;
443         } else { return state.opts[ optionName]; }
444     };

458     /**
459      *  Removes either, all bindings or, all bindings with property paths that
460      *  match a given regex pattern.
461      *
462      *  @param {String|Object} [args.pattern] a regular expression pattern to
463      *      match against, converted to a String using toString() before use
464      *
465      *  @public
466      */
467     this.unbind = function( args ) {
468         var pat = this.definesProperty( args, 'pattern') ?
469             args.pattern.toString() : false;
470         this.each( state.bindings, function( i, n ) {
471             if ( (pat === false) || (i.match( pat) !== null) ) {
472                 delete state.bindings[ i]; } }, this);
473     };

475     /**
476      *  Update all bound DOM nodes with their current model values, if altered.
477      *  <p>
478      *  This function will not update DOM nodes if their model value has not
479      *  changed.
480      *
481      *  @param args.rootNode optional DOM node such that if supplied, only nodes
482      *      under this nodes will be updated.
483      *
484      *  @return {Array} an array of nodes that had their DOM values changed
485      *
486      *  @public
487      */
488     this.updateDOM = function( args ) {
489         var alteredNodes = [];
490         this.each( state.bindings, function( i, n ) {
491             var node = render( {rootNode: this.definesProperty(
492                 args, 'rootNode') ? args.rootNode : undefined,
493                     binding: n, path: i});
494             if ( this.isDefinedNotNull( node) ) {
495                 alteredNodes.push( node); }
496         }, this);
497         return alteredNodes;
498     };

590     /**
591      *  Determines if a given node in the context of a Horn DOM tree is a value
592      *  node or not.
593      *  <p>
594      *  If the node is a value node, the necessary binding information is
595      *  extracted and returned.
596      *
597      *  @param node the DOM node to examine
598      *  @param parentPath the node's parent Horn property path
599      *
600      *  @return <code>false</code> if the given node is not a value node else,
601      *      this function returns an object containing the binding information
602      *      extracted
603      *
604      *  @methodOf Horn.prototype
605      */
606     hasHornBinding: function( node, parentPath ) {
607         var theContained;
608         var nodeName;
609         var path = this.pathIndicator({n: node});
610         var contents = $($(node).contents());
611         var isAdjustingPath = this.isAdjustingPath( path);
612         var cd = {
613             isJSON: this.jsonIndicator({n: node}),
614             node: node};
615         var contentsSize = contents.size();
616         var isEmptyNode = contentsSize === 0;
617         if ( (contentsSize === 1) || (isEmptyNode && !cd.isJSON))  {
618             if ( !isEmptyNode ) { theContained = contents[0]; }
619             if ( cd.isJSON || isAdjustingPath ) {
620                 cd.path = isAdjustingPath ? (parentPath + '-' + path) :
621                     parentPath;
622                 nodeName = node.nodeName.toLowerCase();
623                 cd.isFormElementNode =
624                     (nodeName === 'input') || (nodeName === 'textarea');
625                 cd.isABBRNode = !cd.isFormElementNode &&
626                     (nodeName.toLowerCase() === "abbr");
627                 cd.isTextNode = !cd.isABBRNode && (isEmptyNode ||
628                     (theContained.nodeType === Node.TEXT_NODE));
629 
630                 if ( cd.isFormElementNode || cd.isTextNode || cd.isABBRNode ) {
631                     cd.text = this.getHornDOMNodeValue( {node: node});
632                     return cd;
633                 }
634             }
635         }
636         return false;
637     },

868     /**
869      *  Traverses an object graph and executes a callback function for each
870      *  value encountered.
871      *  <p>
872      *  <code>Object</code> and <code>Array</code> types are considered
873      *  containers of values rather than values themselves.
874      *  <p>
875      *  For any model value, the corresponding equivalent Horn property path is
876      *  constructed and reported to the callback function. An optional property
877      *  path stem may be supplied which will be prepended to any such path.
878      *
879      *  @param value the root of the object graph to traverse
880      *  @param {Function} callback a callback function that will be executed for
881      *      each value of the object graph encountered, it should have the
882      *      following signature: ( path, value, context, propName) where, 'path'
883      *      is the value's Horn path within its container, 'value' is the
884      *      current value found, 'context' is the object in which the value can
885      *      be found and 'key' is the property name within 'context' that the
886      *      value can be found.
887      *  @param {String} [path] an optional property path that will be prefixed
888      *      to every callback reported property path constructed
889      *  @param [context] internal use only - no value required
890      *  @param [propName] internal use only  - no value required
891      *
892      *  @methodOf Horn.prototype
893      */
894     traverse: function( value, callback, path, context, propName ) {
895         if ( (value instanceof Object) || (value instanceof Array) ) {
896             this.each( value, function( k, v ) { this.traverse( v, callback,
897                 path ? (path + '-' + k) : ("-" + k), value, k);
898             }, this);
899         } else { callback( path, value, context, propName); }
900     },

919      *  ( node, path )  - where node is the current node being visited and path

929         this.each( window.$(node).children(), function( i, n ) {
930             if ( fn( n, path) === true ) {
931                 this.visitNodes( n, path, fn); }}, this);
