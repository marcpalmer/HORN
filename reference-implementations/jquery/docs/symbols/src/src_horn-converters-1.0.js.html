  2  *  @fileOverview Provides the <code>HornPatternConverter</code> class that
  3  *      assists in the conversion of Horn model values to their DOM node
  4  *      representation, and vice versa.
  5  *
  6  *  @author <a href="mailto:cdenman@me.com">Chris Denman</a>

  8  *

 10  *
 11  *  @requires JQuery
 12  *  @requires Horn

 15 /**
 16  *  Used to create new HornPatternConverter instances.
 17  *  <p>
 18  *  The HornPatternConverter is a utility 'class' that binds itself to a Horn
 19  *  instance and helps with converting Horn model data.
 20  *  <p>
 21  *  Reusable converter functions are registered to execute when conversion
 22  *  involve property paths that match given expressions.
 23  *
 24  *  @param {Horn} args.horn the Horn instance to bind to
 25  *
 26  *  @constructor
 27  *
 28  *  @test
 29  */
 30 var HornPatternConverter = function (args) {
 31 
 32     /**
 33      *  @private
 34      *  @field
 35      */
 36     var hornInstance;
 37 
 38     /**
 39      *  @private
 40      *  @field
 41      */
 42     var converters;
 43 
 44     /**
 45      *  @private
 46      *  @field
 47      */
 48     var patterns;
 49 
 50     /**
 51      *  Add a named converter.
 52      *  <p>
 53      *
 54      *  @param {String} name the name to associate with the converter
 55      *  @param {Function} converter the converter to add
 56      *
 57      */
 58     this.add = function (name, converter) { converters[name] = converter; };
 59 
 60     /**
 61      *  @private
 62      *  @function
 63      */
 64     var convert = function (args) {
 65         var rv;
 66         hornInstance.each(patterns, function (i, n) {
 67             var match = args.path.match( i);
 68             if (hornInstance.isDefinedNotNull(match) &&
 69                 (match.toString() === args.path)) {
 70                 rv = converters[n](args);
 71                 return false;
 72             }
 73         });
 74         return rv === undefined ? args.value : rv;
 75     };
 76 
 77     /**
 78      *  Retrieve a named converter.
 79      *
 80      *  @param {String} name the name associated with the converter
 81      *
 82      *  @return {Function} the given converter else <code>undefined</code>
 83      *
 84      *  @public
 85      */
 86     this.get = function (name) { return converters[name]; };
 87 
 88     /**
 89      *  Add a pattern, bound to a given named converter.
 90      *
 91      *  @param {String} pattern a Horn property path with optional wildcard '*'
 92      *      characters
 93      *  @param {String} converterName the name of the converter that will handle
 94      *      conversions for the given property path
 95      *
 96      *  @public
 97      */
 98     this.pattern = function (pattern, converterName) {
 99         patterns[pattern] = this.toRegularExpression( converterName);
100     };
101 
102     /**
103      *  Add a regex pattern, bound to a given named converter.
104      *
105      *  @param {String} pattern a valid regular expression that is designed to
106      *      match one or many <code>Horn</code> property paths.
107      *  @param {String} converterName the name of the converter that will handle
108      *      conversions for the given property path
109      *
110      *  @public
111      */
112     this.regexPattern = function (pattern, converterName) {
113         patterns[pattern] = converterName;
114     };
115 
116     /**
117      *  Remove a named converter function.
118      *
119      *  @param {String} name the name associated with the converter function
120      *
121      *  @public
122      */
123     this.remove = function (name) {
124         if ( converters.hasOwnProperty( name) ) { delete converters[name]; }
125     };
126 
127     /**
128      *  Remove a regex pattern (pattern to converter binding).
129      *
130      *  @param {String} pattern the regular expression pattern to remove
131      *
132      *  @public
133      */
134     this.removeRegexPattern = function (pattern) { delete patterns[pattern]; };
135 
136 
137     /**
138      *  Remove a pattern (pattern to converter binding).
139      *
140      *  @param {String} pattern the regular expression pattern to remove
141      *
142      *  @public
143      */
144     this.removePattern = function (pattern) {
145         delete patterns[this.toRegularExpression( pattern)];
146     };
147 
148     /**
149      *  Reset all internal state.
150      *  <p>
151      *  Removes all converters and patterns.
152      *  <p>
153      *  Take a new <code>Horn</code> instance to bind to.
154      *
155      *  @param {Horn} horn the new <code>Horn</code> instance to use
156      *
157      *  @public
158      */
159     this.reset = function ( horn ) {
160         converters = {};
161         patterns = {};
162         hornInstance = horn;
163         hornInstance.option("converter", this.convert);
164     };
165 
166     this.reset(args.horn);
167 };
168 
169 HornPatternConverter.prototype = {
170 
171     /**
172      *  Converts a horn property path with optional wildcard '*' characters into
173      *  a regular expression.
174      *  <p>
175      *  '*' wildcards are converted into '.*' regular expression terms. Array
176      *  indexing and object dereference operators are correctly escaped.
177      *
178      *  @param {String} pattern the Horn property path to convert to a regular
179      *      expression
180      *
181      *  @return {String} a horn property path regular expression
182      *
183      *  @methodOf HornPatternConverter.prototype
184      */
185     toRegularExpression: function( path ) {
186         return path.replace( /([\.\[\]])/g, "\\$1").replace( "*", ".*");
187     }
188 };

190 var hornConverter = new HornPatternConverter({horn: horn});

192 hornConverter.add("Integer",
193     function (args) {
194         if ( args.type !== 'fromJSON' ) {
195             return (args.type === 'fromText' ) ?
196                 parseInt(args.value) : args.value + "";
197         }
198     });
199 
200 hornConverter.add("Boolean",
201     function (args) {
202         if ( args.type !== 'fromJSON' ) {
203             return (args.type === 'fromText') ?
204                 args.value.toLowerCase() === 'true' : args.value + "";
205         }
206     });
