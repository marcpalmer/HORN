 15 $.extend(
 16     Horn.prototype, {
 17         dataNameHorn:           'horn',
 18         dataNameJSON:           'horn-json',
 19 
 20         encodeCSS: function( args ) {
 21             var rv = args.path.replace(
 22                 /(\[(\w+)\])/g, ".$2").replace( /\./g, "-").replace( "/", "");
 23             rv = this.hasPrefix( rv, "-") ? rv.substring(1) : rv;
 24             return rv;
 25         },
 26 
 27         getDataAttr: function( n, name ) {
 28             var rv =  $(n).data( name);
 29             return (this.isDefinedNotNull( rv) === true) ? (rv + "") : undefined;
 30         },
 31 
 32         hasRootIndicator: function( args ) {
 33             var hornDeclaration = this.getDataAttr( args.n, this.dataNameHorn);
 34             var jsonDeclaration = this.getDataAttr( args.n, this.dataNameJSON);
 35             return ((this.isDefinedNotNull( hornDeclaration) === true) &&
 36                 (this.hasPrefix( hornDeclaration, "/") === true)) ||
 37                 ((this.isDefinedNotNull( jsonDeclaration) === true) &&
 38                 (this.hasPrefix( jsonDeclaration, "/") === true));
 39         },
 40 
 41         jsonIndicator: function( args ) {
 42             return this.isDefinedNotNull(
 43                 this.getDataAttr( args.n, this.dataNameJSON)) === true;
 44         },
 45 
 46         pathIndicator: function( args ) {
 47             var hornDeclaration = this.getDataAttr( args.n, this.dataNameHorn);
 48             var jsonDeclaration = this.getDataAttr( args.n, this.dataNameJSON);
 49             if ( jsonDeclaration === 'true' ) { jsonDeclaration = undefined; }
 50             var declaration = this.isDefinedNotNull( hornDeclaration) ?
 51                 hornDeclaration : jsonDeclaration;
 52             return this.isDefinedNotNull( declaration) ?
 53                 this.encodeCSS({path: declaration}) : declaration;
 54         },
 55 
 56         rootNodes: function( args ) {
 57             return $('[data-horn*="/"], [data-horn-json*="/"]');
 58         }
 59     });
