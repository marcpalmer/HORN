 12 /**
 13  *  A Horn delegate implementation that extracts Horn data from DOM elements
 14  *  that possess certain custom CSS classes.
 15  *  <P>
 16  *  Use this implementation if using HTML5 is not an option.
 17  *  <P>
 18  *  Please refer to our online documentation for fuller details
 19  *  <a href="http://horn.io/">http://horn.io/</a>.
 20  *  <P>
 21  *  Use this function to create new <code>HornCSSFeatures</code> instances,
 22  *  thus: <code>var hornCSSFeatures = new HornCSSFeatures();</code>.
 23  *  <P>
 24  *  Set it to use on a horn instance,
 25  *  <code>horn.delegate( hornCSSFeatures);</code>.
 26  *
 27  *  @constructor
 28  *
 29  *  @return {HornCSSFeatures} a newly initialised
 30  *      <code>HornCSSFeatures</code> instance
 31  */
 32 function HornCSSFeatures() {
 33 
 34     /**
 35      *  Determine if a given node possesses a Horn root node indicator
 36      *  (the CSS class 'horn').
 37      *
 38      *  @param {Element} node the node to examine as to declaring a root
 39      *      indicator
 40      *
 41      *  @return <code>true</code> if 'node' does possess a Horn root node
 42      *      indicator, <code>false</code> otherwise.
 43      *
 44      *  @public
 45      */
 46     this.hasRootIndicator = function( node ) {
 47         return $(node).hasClass( HornCSSFeatures.cssRootContext);
 48     };
 49 
 50     /**
 51      *  Extracts and returns the Horn path indicator for a given node.
 52      *  <P>
 53      *  In this implementation, path indicators are CSS class attribute values
 54      *  such as, '_a-0-b-c', that start with the underscore '_' and are formed
 55      *  of words or numbers interspersed with the minus character '-'.
 56      *
 57      *  @param {Element} node the node from which to extract the path indicator
 58      *
 59      *  @return {String|Boolean} if 'node' does have a Horn path indicator, it
 60      *      is returned in <code>String</code> form, else
 61      *      <code>Boolean false</code> is be returned
 62      *
 63      *  @public
 64      */
 65     this.pathIndicator = function( node ) {
 66         var cssPrefixLength = HornCSSFeatures.cssPrefix.length;
 67         var classAttr = $(node).attr( "class");
 68         var path;
 69         if ( classAttr ) {
 70             SMUtils.splitEach( classAttr,
 71                 function( token ) {
 72                     if ( (path === undefined) &&
 73                         SMUtils.hasPrefix( token, HornCSSFeatures.cssPrefix) &&
 74                             (token.length > 1)) {
 75                         path = token.substring( cssPrefixLength);
 76                         return false;
 77                     }
 78                 });
 79         }
 80         return path;
 81     };
 82 
 83     /**
 84      *  Determine if a node declares the JSON indicator.
 85      *  <P>
 86      *  Nodes that declare this indicator are implicitly value nodes and contain
 87      *  literal JSON encoded as the single text element body value child  of
 88      *  declaring elements.
 89      *
 90      *  @param {Element} node the element that may declare the Horn JSON
 91      *      indicator
 92      *
 93      *  @return <code>true</code> if 'node' declares the Horn JSON indicator
 94      *
 95      *  @public
 96      */
 97     this.hasJSONIndicator = function( node ) {
 98         return $(node).hasClass( HornCSSFeatures.cssJSON);
 99     };
100 
101     /**
102      *  Return all the current HTML document's Horn root nodes.
103      *  <P>
104      *  This implementation thus returns all nodes with the CSS attribute
105      *  'class' value '_horn' (regardless of nesting considerations).
106      *
107      *  @return a list of this document's Horn root nodes
108      *
109      *  @public
110      */
111     this.rootNodes = function() {
112         return $("." + HornCSSFeatures.cssRootContext);
113     };
114 }
115 
116 /**
117  *  The expected prefix for Horn property path indicators encoded as CSS
118  *  'class' attribute values.
119  *
120  *  @public
121  */
122 HornCSSFeatures.cssPrefix = '_';
123 
124 /**
125  *  The delimiter used to separate Horn property path tokens encoded as CSS
126  *  'class' attribute values.
127  *
128  *  @public
129  */
130 HornCSSFeatures.cssDelimiter = '-';
131 
132 /**
133  *  The CSS 'class' attribute value, nodes declare to indicate they are the root
134  *  of a data-hierarchy.
135  *
136  *  @public
137  */
138 HornCSSFeatures.cssRootContext = 'horn';
139 
140 /**
141  *  The CSS 'class' attribute value, nodes declare to indicate they contain
142  *  literal JSON data as their single child text element.
143  *
144  *  @public
145  */
146 HornCSSFeatures.cssJSON  = 'data-json';
147 
148 if ( SMUtils.isDefinedNotNull( horn) ) { horn.delegate( new HornCSSFeatures()); }
