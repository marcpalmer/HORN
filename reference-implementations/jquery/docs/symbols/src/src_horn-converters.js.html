1 /**
  2  *  @fileOverview Provides the <code>HornPatternConverter</code> class that
  3  *      assists in mapping <strong>Horn</strong> property paths to
  4  *      corresponding types; and their native JS value <> String value
  5  *      conversions.
  6  *
  7  *  @author <a href="mailto:cdenman@me.com">Chris Denman</a>
  8  *  @author <a href="mailto:marc@anyware.co.uk">Marc Palmer</a>
  9  *
 10  *  @version 1.0
 11  *
 12  *  @requires JQuery
 13  *  @requires Horn
 14  */
 15 
 16 /**
 17  *  Creates a new HornPatternConverter instance which should continue even though
 18  *  we have started a new line.
 19  *
 20  *  @param {Object} args all arguments for this constructor
 21  *  @param {Horn} args.horn the horn instance to which to bind this pattern converter
 22  *      instance
 23  *
 24  *  @constructor
 25  */
 26 HornPatternConverter = function( args ) {
 27 
 28     /**
 29      *  @private
 30      *  @field
 31      */
 32     var instance;
 33 
 34     /**
 35      *  @private
 36      *  @field
 37      */
 38     var converters;
 39 
 40     /**
 41      *  @private
 42      *  @field
 43      */
 44     var patterns;
 45 
 46     $.extend( this, {
 47 
 48         /**
 49          *
 50          *
 51          */
 52         addConverter: function( args ) {
 53             converters[ args.name] = args.converter; },
 54 
 55         /**
 56          *
 57          *
 58          *  @public
 59          */
 60         addPattern: function( args ) {
 61             patterns[ args.pattern] = args.converterName; },
 62 
 63         /**
 64          *
 65          *  @return
 66          *
 67          *  @public
 68          */
 69         convert: function( args ) {
 70             var rv;
 71             instance.each( patterns, function( i, n ) {
 72                 var match = args.path.match( i);
 73                 if ( instance.isDefinedNotNull( match) &&
 74                     (match.toString() === args.path) ) {
 75                     rv = converters[ n]( args);
 76                     return false;
 77                 }
 78             });
 79             return rv === undefined ? args.value : rv;
 80         },
 81 
 82         /**
 83          *
 84          *  @return
 85          *
 86          *  @public
 87          */
 88         getConverter: function( args) { return converters[ args.name]; },
 89 
 90         /**
 91          *
 92          *
 93          *  @public
 94          */
 95         removeConverter: function( args ) { delete converters[ args.name]; },
 96 
 97         /**
 98          *
 99          *
100          *  @public
101          */
102         removePattern: function( args ) { delete patterns[ args.name]; },
103 
104         /**
105          *
106          *
107          *  @public
108          */
109         reset: function( args ) {
110             instance = undefined;
111             converters = {};
112             patterns = {};
113         }
114     });
115 
116     reset();
117     instance = args.horn;
118     instance.option( "converter", this.convert);
119 };
120 
121 var hornConverter = new HornPatternConverter({horn: horn});
122 
123 hornConverter.addConverter( {name: "IntegerConverter", converter:
124     function( args ) {
125         return args.type === 'fromText' ?
126             parseInt( args.value) : args.value + "";
127 }});
128 
129 hornConverter.addConverter( {name: "BooleanConverter", converter:
130     function( args ) {
131         return args.type === 'fromText' ?
132             args.value.toLowerCase() === 'true' : args.value + "";
133 }});
