1 /**
  2  *  @fileOverview Provides the <code>HornPatternConverter</code> class that
  3  *      assists in the conversion of Horn model values to their DOM node
  4  *      representation, and vice versa.
  5  *
  6  *  @author <a href="mailto:cdenman@me.com">Chris Denman</a>
  7  *  @author <a href="mailto:marc@anyware.co.uk">Marc Palmer</a>
  8  *
  9  *  @version 1.0
 10  *
 11  *  @requires JQuery
 12  *  @requires Horn
 13  */
 14 
 15 /**
 16  *  Used to create new HornPatternConverter instances.
 17  *
 18  *  @param {Horn} args.horn the Horn instance to bind to
 19  *
 20  *  @constructor
 21  */
 22 var HornPatternConverter = function(args) {
 23 
 24     /**
 25      *  @private
 26      *  @field
 27      */
 28     var instance;
 29 
 30     /**
 31      *  @private
 32      *  @field
 33      */
 34     var converters;
 35 
 36     /**
 37      *  @private
 38      *  @field
 39      */
 40     var patterns;
 41 
 42     /**
 43      *
 44      *
 45      */
 46     this.add = function (name, converter) {
 47         converters[name] = converter;
 48     };
 49 
 50     /**
 51      *
 52      *
 53      *  @public
 54      */
 55     this.pattern = function (pattern, converterName) {
 56         patterns[pattern] = converterName;
 57     };
 58 
 59     /**
 60      *
 61      *  @return
 62      *
 63      *  @public
 64      */
 65     this.convert = function (args) {
 66         var rv;
 67         instance.each(patterns, function (i, n) {
 68             var match = args.path.match(i);
 69             if (instance.isDefinedNotNull(match) && (match.toString() === args.path)) {
 70                 rv = converters[n](args);
 71                 return false;
 72             }
 73         });
 74         return rv === undefined ? args.value : rv;
 75     };
 76 
 77     /**
 78      *
 79      *
 80      *  @public
 81      */
 82     this.remove = function (name) {
 83         delete converters[name];
 84     };
 85 
 86     this.get = function (name) {
 87         return converters[name];
 88     };
 89 
 90     /**
 91      *
 92      *
 93      *  @public
 94      */
 95     this.removePattern = function (pattern) {
 96         delete patterns[pattern];
 97     };
 98 
 99     /**
100      *  Reset all
101      *
102      *  @public
103      */
104     this.reset = function (args) {
105         instance = undefined;
106         converters = {};
107         patterns = {};
108     };
109 
110     this.reset();
111     instance = args.horn;
112     instance.option("converter", this.convert);
113 };
114 
115 var hornConverter = new HornPatternConverter({horn: horn});
116 
117 hornConverter.add("Integer",
118     function (args) {
119         return args.type === 'fromText' ?
120             parseInt(args.value) : args.value + "";
121     });
122 
123 hornConverter.add("Boolean",
124     function (args) {
125         return args.type === 'fromText' ?
126             args.value.toLowerCase() === 'true' : args.value + "";
127     });
